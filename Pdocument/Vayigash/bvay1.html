גַּמְלִיאֵל, נוֹהֲגִין הָיוּ בֵית אַבָּא, נוֹתְנִין פֵּאָה אַחַת לַזֵּיתִים שֶׁהָיוּ לָהֶם בְּכָל רוּחַ, וְלֶחָרוּבִין כָּל הָרוֹאִין זֶה אֶת זֶה. רַבִּי אֶלִיעֶזֶר בְּרַבִּי צָדוֹק אוֹמֵר מִשְּׁמוֹ, אַף לֶחָרוּבִין שֶׁהָיוּ לָהֶם בְּכָל הָעִיר: ה. הַזּוֹרֵעַ אֶת שָׂדֵהוּ מִין אֶחָד, אַף עַל פִּי שֶׁהוּא עוֹשֵׂהוּ שְׁתֵּי גְרָנוֹת, נוֹתֵן פֵּאָה אֶחָת. זְרָעָהּ שְׁנֵי מִינִין, אַף עַל פִּי שֶׁעֲשָׂאָן גֹּרֶן אַחַת, נוֹתֵן שְׁתֵּי פֵאוֹת. הַזּוֹרֵעַ אֶת שָׂדֵהוּ שְׁנֵי מִינֵי חִטִּין, עֲשָׂאָן גֹּרֶן אַחַת, נוֹתֵן פֵּאָה אַחַת שְׁתֵּי גְרָנוֹת, נוֹתֵן שְׁתֵּי פֵאוֹת: ו. מַעֲשֶׂה שֶׁזָּרַע רַבִּי שִׁמְעוֹן אִישׁ הַמִּצְפָּה לִפְנֵי רַבָּן גַּמְלִיאֵל, וְעָלוּ לְלִשְׁכַּת הַגָּזִית וְשָׁאָלוּ. אָמַר נַחוּם הַלַּבְלָר, מְקֵבָּל אֲנִי מֵרַבִּי מְיָאשָׁא, שֶׁקִּבֵּל מֵאַבָּא, שֶׁקִּבֵּל מִן הַזּוּגוֹת, שֶׁקִּבְּלוּ מִן הַנְּבִיאִים, הֲלָכָה לְמשֶׁה מִסִינַי, בְּזוֹרֵעַ אֶת שָׂדֵהוּ שְׁנֵי מִינֵי חִטִּין, אִם עֲשָׂאָן גֹּרֶן אַחַת, נוֹתֵן פֵּאָה אַחַת שְׁתֵּי גְרָנוֹת, נוֹתֵן שְׁתֵּי פֵאוֹת: ז. שָׂדֶה שֶׁקְּצָרוּהָ גוֹיִם, קְצָרוּהָ לִסְטִים, קִרְסְמוּהָ נְמָלִים, שְׁבָרַתָּהּ הָרוּחַ אוֹ בְהֵמָה, פְּטוּרָה. קָצַר חֶצְיָהּ וְקָצְרוּ לִסְטִים חֶצְיָהּ, פְּטוּרָה, שֶׁחוֹבַת הַפֵּאָה בַּקָּמָה: ח. קְצָרוּהָ לִסְטִים חֶצְיָהּ וְקָצַר הוּא חֶצְיָהּ, נוֹתֵן פֵּאָה מִמַּה שֶּׁקָּצַר. קָצַר חֶצְיָהּ וּמָכַר חֶצְיָהּ, הַלּוֹקֵחַ נוֹתֵן פֵּאָה לַכֹּל. קָצַר חֶצְיָהּ וְהִקְדִּישׁ חֶצְיָהּ, הַפּוֹדֶה מִיַּד הַגִּזְבָּר, הוּא נוֹתֵן פֵּאָה לַכֹּל:

    ר' עובדיה מברטנורא

    (א) ואלו מפסיקין. בין שדה לשדה ויהיו נחשבים כשתי שדות לענין שאם מניח פאה באחת מהן על חברתה לא עלתה לו לפאה, דכתיב (ויקרא יט) פאת שדך, שלא יניח פאה משדה לחברתה: הנחל. נהר: והשלולית. אמת המים, שמחלקת שלל לאגפיה, שאמות אחרות של מים שותות ממנה: ודרך היחיד. ארבע אמות: ודרך הרבים. שש עשרה אמה, ומשום סיפא נקט לה, דבעי למתני סיפא שכל אלו אין מפסיקין באילן. וקא משמע לן דאפילו דרך הרבים שהיא רחבה שש עשרה אמה אינה מפסקת באילן: שביל הרבים ושביל היחיד. שביל זוטר טובא דשקיל כרעא ומנח כרעא, ואשמועינן דאפילו שביל דרבים אם הוא קבוע לימות החמה ולימות הגשמים, כלומר, שמהלכין בו אפילו בזמן שהשדות זרועות בימות הגשמים, מפסיק בין השדות הזרועות. ואי לא, לא מפסיק: הבור. שדה שאינה זרועה. והאדמה לא תשם (בראשית מז), מתרגמינן וארעא לא תבור: והניר. חרישה כמו נירו לכם ניר (ירמיה ד): וזרע אחר. כגון ששתי השדות זרועות חטים וביניהם קרקע זרוע מין אחר. ושיעור רוחב הבור והניר וזרע אחר, כדי שלשה תלמים של מענית המחרישה: והקוצר לשחת מפסיק. דקסבר כל לשחת לאו קצירה היא דלא חשבינן ליה אתחלתא דקצירה. שחת, תבואה שלא הביאה שליש וקוצרים אותה להאכיל לבהמות. ואין הלכה כרבי מאיר: (ב) אמת המים שאינה יכולה להקצר כאחת. שהיא רחבה כל כך עד שהעומד באמצע אינו יכול לקצור מכאן ומכאן: ר' יהודה אומר מפסקת. ופליג אתנא קמא דאמר לעיל על שלולית דהיינו אמת המים דבכל ענין מפסיק. ור' יהודה סבר דאם יכולה להקצר כאחת, אינה מפסקת. והלכה כר' יהודה: וכל ההרים אשר במעדר יעדרון. הר שחודו זקוף ואין בקר בכליו יכולין לעבור שם ומפסיק בין שתי השדות: הוא נותן פאה לכל. כלומר, נותן פאה אחת לכל שתי השדות ולא חשיב הפסק הואיל והם נעדרים במעדר, כלומר, שבני אדם חופרים אותו הר בכלי שחופרים בו את הקרקע, אין זה הפסק, שהרואה אומר אין זה קרקע בור, אלא למחר חופרים אותו במעדר וזורעים אותו, ושתי השדות הם אחת: (ג) הכל מפסיק לזרעים. מפרש בגמ' בירושלמי דהכל לאתויי סלע שהיה עובר על פני כל השדה. אם צריך הוא לעקור את המחרישה מצד זה כדי לתת אותה לצד זה, מפסיק: אלא גדר. גבוה עשרה טפחים: שער כותש. ענפי האילן קרויין שער, ופירוש שער כותש, שהנופות של אילנות מתערבין זה עם זה למעלה מן הגדר, כעלי זה התקוע במכתש: (ד) כל הרואים זה את זה. כל זמן שהעומד אצל אחד מהאילנות יכול לראות האילן האחר אף על פי שיש ביניהן גדר אינו מפסיק ולוקח פאה מאילן אחד על חבירו: לזיתים שהיו להם בכל רוח. פאה אחת היו נותנים על כל הזיתים שהיו להם לצד מזרחה של עיר, ופאה אחרת על כל הזיתים שהיו להם לצד מערבה, וכן לארבע רוחות: שהיו להם בכל העיר. אין נותנים כי אם פאה אחת לכולן, אף על פי שאין רואין זה את זה. ואין הלכה כר' אליעזר ברבי צדוק שאמר משמו, אבל הלכה כרבן גמליאל: (ה) שני מיני חטין. כגון שחמתית ולבנה: (ו) הלבלר. הסופר: הזוגות. הם כל אותן שנים שנים הנזכרים בפרק קמא דמסכת אבות, שקבלו שנים מפי שנים עד מפי שמעון הצדיק: (ז) שדה שקצרוה גויים. לעצמן, ולא שקצרו פועלי גויים לישראל, דאז הוי כאלו קצרוה ישראל: קרסמוה נמלים. דרך הנמלה לחתוך קנה השבולת מלמטה, ולזה קורים קרסום לשון יכרסמנה חזיר מיער (תהלים פ): פטורה. דכתיב גבי פאה ובקוצרכם, עד שתהיו אתם הקוצרים: שחובת הפאה בקמה. כלומר, חובת הפאה של זה הניחה בקמה, ונהי דאם כלה שדה חזרה פאה לעומרים, וחייב להפריש פאה מן העומרים, הני מילי היכא דכלה הוא, אבל הכא הוא לא כלה: (ח) הלוקח נותן פאה לכל. שחובת הפאה ממה שקצר נשאר בזה החצי שקנה הלוקח, והוי כאלו לא מכר לו אלא מה שנשאר בשדה אחר שיוציא ממנה הפאה שראוי להוציא מאותה שדה. וכן הפודה מיד הגזבר, מוציא מן החצי שפדה, הפאה הראויה לשדה כולה:

    גמרא

    ברכות דף ד' ע''א

    תַּנְיָא אַבָּא בִּנְיָמִין אוֹמֵר עַל שְׁנֵי דְבָרִים הָיִיתִי מִצְטָעֵר כָּל יָמַי עַל תְּפִלָּתִי שֶׁתְּהֵא לִפְנֵי מִטָּתִי וְעַל מִטָּתִי שֶׁתְּהֵא נְתוּנָה בֵּין צָפוֹן לְדָרוֹם. עַל תְּפִלָּתִי שֶׁתְּהֵא לִפְנֵי מִטָּתִי. מַאי לִפְנֵי מִטָּתִי אִילֵימָא לִפְנֵי מִטָּתִי מַמָּשׁ וְהָאָמַר רַב יְהוּדָה אָמַר רַב וְאִי תֵּימָא רִבִּי יְהוֹשֻׁעַ בֶּן לֵוִי מְנַיִן לְמִתְפַּלֵּל שֶׁלֹּא יְהֵא דָבָר חוֹצֵץ בֵּינוֹ לְבֵין הַקִּיר שֶׁנֶּאֱמַר (ישעי' ל''ח) וַיַּסֵּב חִזְקִיָּהוּ פָנָיו אֶל הַקִּיר וַיִּתְפַּלֵּל לֹא תֵּימָא לִפְנֵי מִטָּתִי אֶלָא אֵימָא סָמוּךְ לְמִטָתִי. וְעַל מִטָּתִי שֶׁתְּהֵא נְתוּנָה בֵּין צָפוֹן לְדָרוֹם דְאָמַר רִבִּי חָמָא בְּרִבִּי חֲנִינָא אָמַר רִבִּי יִצְחָק כָּל הַנּוֹתֵן מִטָּתוֹ בֵּין צָפוֹן לְדָרוֹם הַוְיָין לֵיהּ בָּנִים זְכָרִים שֶׁנֶּאֱמַר (תהלים י''ז) וּצְפוּנְךָ תְּמַלֵּא בִטְנָם יִשְׂבְּעוּ בָנִים. רַב נַחְמָן בַּר יִצְחָק אָמַר אַף אֵין אִשְׁתּוֹ מַפֶּלֶת נְפָלִים כְּתִיב הָכָא וצְפוּנְךָ תְּמַלֵּא בִטְנָם. וּכְתִיב הָתָם וַיִּמְלְאוּ יָמֶיהָ לָלֶדֶת וְהִנֵּה תוֹמִים בְּבִטְנָהּ. תַּנְיָא אַבָּא בִּנְיָמִין אוֹמֵר שְׁנַיִם שֶׁנִּכְנְסוּ לְהִתְפַּלֵּל וְקָדַם אֶחָד מֵהֶם לְהִתְפַּלֵּל וְלֹא הִמְתִּין אֶת חֲבֵירוֹ וְיָצָא טוֹרְפִין לוֹ תְּפִלָּתוֹ בְּפָנָיו. שֶׁנֶּאֱמַר (איוב י''ח) טוֹרֵף נַפְשׁוֹ בְּאַפּוֹ הַלְמַעַנְךָ תֵּעָזֵב אָרֶץ. וְלֹא עוֹד אֶלָּא שֶׁגּוֹרֵם לִשְׁכִינָה שֶׁתִּסְתַּלֵּק מִיִּשְׂרָאֵל שֶׁנֶּאֱמַר (שם) וְיֶעְתַּק צוּר מִמְּקוֹמוֹ וְאֵין צוּר אֶלָּא הַקָּדוֹשׁ בָּרוּךְ הוּא שֶׁנֶּאֱמַר (דברים ל''ב) צוּר יְלָדְךָ תֶּשִׁי. וְאִם הִמְתִּין לוֹ מַה שְׂכָרוֹ אָמַר רִבִּי יוֹסֵי בְּרִבִּי חֲנִינָא זוֹכֶה לִבְרָכוֹת הַלָּלוּ שֶׁנֶּאֱמַר (ישעי' מ''ח) לוּא הִקְשַׁבְתָּ לְמִצְוֹתָי. וַיְהִי כַנָּהָר שְׁלוֹמֶךָ וְצִדְקָתְךָ כְּגַלֵּי הַיָם. וַיְהִי כַחוֹל זַרְעֶךָ וְצֶאֱצָאֵי מֵעֶיךָ וְגוֹ':

    תרגום הגמרא

    למדנו בברייתא: אבא בנימין אומר: על שני דברים הייתי מצטער (כלומר דואג ומשתדל) כל ימי- על תפלתי שתהא לפני מטתי, ועל מטתי שתהא נתונה בין צפון לדרום. "על תפלתי שתהא לפני מטתי" – מה כוונתו 'לפני מטתי'? אם נאמר לפני מטתי ממש, והרי אמר רב יהודה משמו של רב, ויש אומרים- רבי יהושע בן לוי: מנין לנו שהמתפלל צריך להשתדל שלא יהיה דבר חוצץ בינו לבין הקיר? שנאמר (ישעיה ל"ח) "ויסב חזקיה פניו אל הקיר ויתפלל?!" ואם כן אין להתפלל לפני המטה?! אלא צריך לומר, שכוונתו של אבא בנימין היתה: סמוך למטתי (זאת אומרת שלא עשה כלום מאז שקם ממטתו עד שהתפלל). "ועל מטתי שתהא נתונה בין צפון לדרום"- שאמר ר' חמא בן רבי חנינא משמו של רבי יצחק: כל המעמיד מטתו בין צפון לדרום, כלומר אורך המטה בקו צפון דרום- יהיו לו בנים זכרים, שנאמר (תהלים י"ז) "וצפונך תמלא בטנם ישבעו בנים", ('וצפונך'- לשום צפון ובסוף הפסוק נאמר 'ישבעו בנים'.), רב נחמן בר יצחק הוסיף ואמר: אף אין אשתו מפלת נפלים, שכן כתוב כאן "וצפונך תמלא בטנם" כלומר תמלא ימי הריונה, וכתוב בפסוק אחר (בראשית כ"ה) "וימלאו ימיה ללדת והנה תומים בבטנה". למדנו בברייתא: אבא בנימין אומר: שנים שנכנסו להתפלל וקדם אחד מהם וסיים להתפלל ולא המתין לחברו ויצא- טורפים לו המלאכים הרעים את תפלתו בפניו ומונעים אותה מלהגיע לפניו יתברך, שנאמר (איוב י"ח) "טורף נפשו באפו הלמענך תעזב ארץ" (כי בדורות הקדומים היו בתי הכנסת מחוץ לעיר והיה פחד וסכנה לשהות בהם לבד), ולא עוד אלא שגורם לשכינה שתסתלק מישראל, שנאמר (שם) "ויעתק צור ממקומו" ואין צור אלא הקב"ה, שנאמר (דברים ל"ב) "צור ילדך תשי". ואם המתין לו מה שכרו? אמר רבי יוסי בן רבי חנינא: זוכה לברכות הללו: שנאמר (ישעיה מ"ח) "לוּא הקשבת (הוא לשון המתנה) למצותי (כלומר בשביל מצותי שצויתי לגמול חסד) ויהי כנהר שלומך וצדקתך כגלי הים, ויהי כחול זרעך וצאצאי מעיך" וגו'.

    Mishná

    Tratado Peá, Capítulo 2

    (1) Los siguientes dividen los campos con relación a Peá: Un arroyo, un canal, una vía privada, una vía pública un sendero público, un sendero privado de uso continuo tanto en días de sol como en días de lluvia, un campo sin cultivar, un campo arado y una zona con diferente cultivo. Quien cosecha sólo para alimentar animales crea una división según Rabí Meir. Para los demás Sabios no cabe división a menos que are allí. (2) Un canal desde donde no se puede cosechar los dos campos simultáneamente, Rabí Yehudá opina que divide. Los montículos que precisan azada, a pesar de que el buey no puede pasar con su arado, no ocasionan división y, por tanto, aparta una Peá por todo. (3) Todo lo anterior divide en el caso de sembrados, pero en el caso de arboledas sólo un cerco divide. En caso de haber “cabello cortado” no hay división, y aparta una Peá por todo. (4) Con respecto al algarrobo, siempre que se vea de uno a otro. Rabán Gamliel dijo: Acostumbraban en casa de mi padre a dar una Peá por los olivares en cada uno de los puntos cardinales, y por los algarrobos si se veían de uno a otro. Rabí Eliézer hijo de Rabí Tzadok dijo en su nombre: Igual se hacía con los algarrobos que había en toda la ciudad. (5) Quien siembra su campo con una especie, aunque la almacene en dos graneros, dará una sola Peá. Si siembra dos especies, aunque las almacene en un solo granero, dará dos Peá. Quien siembra su campo con dos clases de trigo, si las almacenó en un solo granero, separará una Peá, mas si las guardó en dos graneros, dará dos Peá. (6) Cuentan que sembró Rabí Shimón de Mitzpá en presencia de Rabán Gamliel y ambos fueron al Gran Tribunal a consultar. Dijo Najum el escriba: Me fue enseñada una ley a través de Rabí Meyashá, quien la aprendió de Abá, quien, a su vez, la aprendió de Las Parejas, y estos, de los Profetas, una disposición de Moshé, desde el Sinai: Quien siembra su campo con dos tipos de trigo y los almacena en un solo granero, aparta, solamente, una Peá; en dos graneros, dos Peá. (7) Un campo cosechado por gentiles, o por ladrones, o roído por hormigas, o que haya sido destruido por el viento o el ganado, queda eximido de Peá. Si fue cosechado la mitad, y ladrones cosecharon la otra mitad, igualmente queda eximido por cuanto que la obligación de apartar Peá es al finalizar la cosecha. (8) Si los ladrones cosecharon la mitad y el propietario la otra mitad, dará Peá de lo que él cosechó. Si cosechó la mitad y vendió la otra mitad, el comprador debe dar Peá por todo. Si cosechó la mitad y consagró la otra mitad, quien la rescata de manos del tesorero, dará Peá por todo.

    Rabí Ovadiá de Bartenura

    1. LOS SIGUIENTES DIVIDEN. Entre un campo y otro, siendo estos considerados como dos campos, de forma que si apartó en uno la Peá por los dos, no le valió al segundo, tal como está escrito: “La Peá de tu campo” (Levítico 8:16), o sea, no apartar la Peá de un campo para otro. UN ARROYO. Un río. UN CANAL. Se refiere a un canal que divide su caudal a ambos márgenes, y a su vez otros canales se nutren de él. UNA VÍA PRIVADA. De cuatro codos de ancho. UNA VÍA PÚBLICA. De dieciséis codos de ancho. Este ejemplo fue mencionado por la necesidad de indicarnos al final del capítulo que en campos de árboles una vía de este ancho no ocasiona una separación. UN SENDERO PÚBLICO, UN SENDERO PRIVADO. Son mucho más pequeños que los anteriores, siendo estos del ancho de un caminante. Y nos enseña que un sendero, aún público, si es usado tanto en verano como en invierno, o sea, que es utilizado incluso cuando los campos están sembrados, provoca una separación, y en caso contrario, no. UN CAMPO SIN CULTIVAR. Porque está escrito: “Que la tierra no quede desolada” (Génesis 47), donde la traducción al arameo indica que no hay cultivo. UN CAMPO ARADO. Como en “Aren vuestro surco” (Jeremías 4:3). DIFERENTE CULTIVO. Como por ejemplo, dos campos sembrados de trigo, separados por otro cultivo. La medida mínima del ancho de un campo sin cultivar, de un campo arado y del de cultivo distinto, debe ser una franja de tres surcos de arado. QUIEN COSECHA SÓLO PARA ALIMENTAR ANIMALES. La idea es que lo que se cosecha para este fin no es conceptuado como parte de la cosecha. Este heno es un cereal que no llegó al tercio de su maduración y que es usado para alimentar animales. La ley no es como Rabí Meir. 2. UN CANAL DESDE DONDE... Es tan ancho que quien se encuentra en su centro no alcanza a cosechar ni un lado ni otro. RABÍ YEHUDÁ OPINA QUE DIVIDE. Difiere de la opinión del primer exponente de la Mishná quien decía que un canal siempre divide. Rabí Yehudá opina que si desde ese canal puede cosechar los dos campos al mismo tiempo, no divide. La ley se dictamina según la opinión de Rabí Yehudá. LOS MONTÍCULOS QUE PRECISAN AZADA... Un montículo muy empinado, el cual los bueyes no pueden arar, divide dos campos. APARTA UNA PEÁ POR TODO. Es decir, por los dos campos y ese monte no divide siendo que se puede arar manualmente. Además, quien lo ve advierte que no es un campo sin cultivar, sino que en cualquier momento puede ser arado y así los dos campos se consideran uno. 3. TODO LO ANTERIOR... Está explicado en el Talmud Yerushalmi que la expresión “todo” viene a incluir el caso de una roca que atraviesa el campo de un lado a otro. Si, para arar, debe levantar el arado y pasarlo al otro lado de la roca, esta los divide. SÓLO UN CERCO DIVIDE. De diez puños de altura. “CABELLO CORTADO”. Las ramas del árbol son denominadas figurativamente “cabello”, y la expresión “cabello cortado” se refiere a que las ramas de los árboles sobrepasan la altura del cerco y se entrecruzan en las copas, como las hojas que se han trabado en un mortero. 4. SIEMPRE QUE SE VEA DE UNO A OTRO. Es decir, que quien se pare junto a uno de estos árboles y pueda ver el otro, aunque entre ellos haya un cerco, este no los divide. La Peá de uno vale para el otro. LOS OLIVARES... Apartaban una Peá por todos los olivos del oriente de la ciudad y otra por todos los olivos del lado occidental, y así en todos los lados. QUE HABÍA EN TODA LA CIUDAD. Daban una sola Peá por todos ellos, a pesar de que no se ven unos a otros. La ley no concuerda con Rabí Eliézer hijo de Rabí Tzadok, sino con Rabán Gamliel. 5. DOS CLASES DE TRIGO. Como el moreno y el blanco. 6. EL ESCRIBA. El escribano. LAS PAREJAS. Son todas las mencionadas en el primer capítulo del Tratado de Avot, las cuales recibieron la tradición, cada pareja de su precedente, desde Shimón Hatzadik. 7. UN CAMPO COSECHADO POR GENTILES. Para ellos mismos, no que cosecharon unos trabajadores gentiles en beneficio de judíos, en cuyo caso sería como si los mismos judíos lo cosecharan. ROÍDO POR HORMIGAS. La hormiga suele comer la parte inferior de la espiga. Esto es expresado con el verbo “roer”, como en: “Roída por el jabalí del bosque” (Salmos 80:14). QUEDA EXIMIDO. Siendo que el versículo dice en el tema de Peá: “Cuando ustedes cosechen”, esto es, sólo cuando lo hagan ustedes. LA OBLIGACIÓN DE APARTAR PEÁ... O sea, que la obligación recae al finalizar, y a pesar de que al finalizar habría que apartarla de las gavillas, esto es solo en el supuesto de que él mismo finalizó, no otros. 8 EL COMPRADOR DEBE DAR PEÁ POR TODO. La obligación de apartar Peá por lo cosechado pasa a la mitad que le compraron, y se considera como si le hubiese vendido solamente el resto de la cosecha después de haber separado la Peá correspondiente a ese campo. Así también, quien rescata de la mano del tesorero, de su mitad separa Peá correspondiente a todo el campo.

    Guemará

    Tratado Berajot 5b

    Estudiamos en una Baraita: Abá Binyamín dice: Por dos cosas me preocupé toda mi vida: Que mi plegaria estuviera delante de mi cama y que mi cama estuviera colocada entre el norte y el sur. ¿Qué significa: “delante de mi cama”? ¿Acaso lo debemos entender literalmente, “delante”? ¿Acaso no dijo Rab Yehudá en nombre de Rab (algunos dicen que era Rabí Yehoshúa Ben Leví): De dónde sabemos que nada debe interponerse entre la pared y la persona que reza? El versículo nos dice: “Y Yejezkel dirigió su rostro hacia la pared y rezó” (Isaías 38:2). Por tanto, en lugar de “delante de mi cama”, habría que decir: “cerca de mi cama”. “Y que mi cama estuviera colocada entre el norte y el sur”, por aquello que dijo Ribí Jamá hijo de Rabí Janiná en nombre de Rabí Yitzjak: “Todo aquel que ubique su cama entre el norte y el sur tendrá hijos varones”, tal como versa: “Las cosas (buenas) que ocultaste colmarán sus vientres, se saciarán de hijos” (Salmos 17:14). Rab Najmán Bar Yitzjak dijo: “además, su esposa no sufrirá abortos”. Aquí está escrito: “Las cosas que ocultaste colmarán sus vientres” y en otro lugar: “Y culminaron los días de su gestación, y he aquí que tenía mellizos en su vientre” (Génesis 25:24). Estudiamos en una Baraita: Abá Binyamín dice: Cuando dos hombres entran a una Bet-Hakneset a rezar, y uno de ellos acaba antes y sale sin esperar al compañero, le arrojan su rezo a la cara, tal como versa: “Tú, que te desgarras en tu ira, ¿por ti debe ser abandonada la tierra?” (Job 18:4). Además, provoca que se aleje la Presencia Divina de Israel, como está dicho: “Y arrancarás la Roca de su lugar” (ibíd). Y la “Roca” no se refiere sino al Santo, Bendito Él, como está dicho: “La Roca que te dio a luz, olvidaste” (Deuteronomio 32:18). Y si lo espera, ¿cuál es su recompensa? Rabí Yosé hijo de Rabí Janiná dijo: Es recompensado con las bendiciones siguientes, según lo escrito: “¡Oh, si hubieras escuchado mis mandamientos! Tu paz sería como la de un río y tu rectitud como la de las olas del mar, y sería como la arena tu simiente, y la descendencia de tus entrañas...” (Isaías 48:18-19).

    Rashí

    CERCA DE MI CAMA. O sea, toda mi vida me cuidé de no ocuparme con ningún trabajo, ni de estudiar Torá, desde que me levantaba de la cama hasta después de recitar la Shemá y rezar. ENTRE EL NORTE Y EL SUR. La cabecera al norte y los pies al sur. Me parece que la Presencia Divina se manifiesta en el este o el oeste, por eso lo correcto es llevar a cabo las relaciones conyugales en otros puntos cardinales. LAS COSAS (BUENAS) QUE OCULTASTE. Esta expresión en hebreo alude al norte, y en el final de ese versículo se dice: “Se saciarán de hijos”. COLMARÁN SUS VIENTRES. Completará los días de su gestación. QUE TE DESGARRAS EN TU IRA. Esto fue dicho sobre quien provoca que su “alma” sea desgarrada. La expresión “alma” se refiere al rezo, siendo que el versículo dice: Derramaré mi “alma” delante del Eterno. ¿POR TI DEBE SER ABANDONADA LA TIERRA? ¿Acaso pensaste que por marcharte tú, la Presencia Divina iba a abandonar a tu compañero que rezaba ante Él? SI HUBIERAS ESCUCHADO. Con el sentido de “esperar”. MIS MANDAMIENTOS. Los mandamientos en los cuales ordené prodigar el bien.

    (Comentario sobre el Zóhar)

    Rabí Yehudá y Rabí Yosé coincidieron en la aldea Janán. Mientras se hospedaban en la posada, llegó un hombre precedido por un asno cargado, que se introdujo en la vivienda. Mientras tanto, dijo Rabí Yehudá a Rabí Yosé: ¿Acaso no aprendimos que el rey David, dormía como lo hace un caballo, esto es, muy poco tiempo? Entonces, ¿cómo es que se levantaba a medianoche, siendo que el tiempo que dedicaba a dormir era muy corto, no más de sesenta respiraciones? ¡Es muy poco tiempo, ni siquiera hubiese podido dormir un tercio de la noche! Le contestó: Cuando entraba la noche acostumbraba a sentarse con los más grandes de su casa y hablaba de leyes, se sumergía en el estudio de la Torá, y después se marchaba a dormir hasta llegada la medianoche. Se levantaba en aquella hora y se entregaba a servir a su Creador con cánticos y alabanzas. Después de esto, les comentó aquel hombre: tal como habéis hablado, este es el secreto con el que se explica que el rey David vive eternamente. El rey David se cuidó toda su vida de no probar el gusto de la muerte, por cuanto que, el sueño es una sesentava parte de la muerte. El rey David, que aún vive, nunca durmió el tiempo de sesenta respiraciones, puesto que hasta sesenta menos una, la persona se considera viva, de allí en adelante, prueba el gusto de la muerte y se apodera de él el Espíritu de la Impureza. La razón por la cual el rey David se cuidaba de no probar el gusto de la muerte, para que no se pudiera apoderar de él el Espíritu de la Impureza, era porque sabía que sesenta menos una respiraciones, es el fundamento de la vida de arriba. Hasta sesenta respiraciones que son sesenta respiraciones de arriba, de cuyo fundamento depende la vida. De allí hacia abajo se encuentra el fundamento de la muerte. Por esta razón, David, el rey, calculaba el tiempo de la medianoche para poder permanecer en la vida y para que no pudiera gobernar sobre él el sabor de la muerte. Al llegar la medianoche exactamente, se ponía en pie. Al despertar la medianoche, momento en el cual la Corona Sagrada también despertaba, era necesario que no encontrara a David ligado a otro lugar, el lugar de la muerte. Cuando la noche se divide y se despierta la Santidad Suprema, y el hombre que duerme sobre su lecho no se despierta para reparar en el Honor de su Creador, entonces, se liga al fundamento de la muerte y se apega a otro lugar. Por esto, se levantaba el rey David, para otorgar el Honor debido a su Creador, vivir ante Quien siempre vive. Nunca dormía lo suficiente hasta probar el gusto de la muerte, por esta razón dormía como lo hace un caballo, sesenta respiraciones sin completar (cincuenta y nueve). Se acercaron Rabí Yehudá y Rabí Yosé y lo besaron. Le preguntaron: ¿Cuál es tu nombre? Les contestó: Jizquiyahu. Le dijeron: ¡Que seas bienaventurado y que sea fortalecida tu Torá!

    Halajá

    Rambam, Leyes sobre el Arrepentimiento, capítulo 1

    (1) Todos los preceptos de la Torá, tanto sean de carácter activo como pasivo, si son transgredidos, ya sea de forma voluntaria o involuntaria, al percatarse y arrepentirse el transgresor, deberá confesarse delante del Eterno, Bendito Sea, tal como versa: “Hombre o mujer que cometieren... y se confesarán del pecado que hicieron...” (Números 5:6-7). Esto hace alusión a una confesión verbal. Esta confesión es uno de los preceptos activos. (2) ¿En qué consiste la confesión? Deberá decir: “Por favor, Eterno: he pecado, he transgredido, Te he faltado, he hecho tal y tal cosa... Me arrepiento y me avergüenzo de mis actos. Nunca más lo volveré a hacer”. Esto representa lo más esencial de la confesión. No obstante, cuanto más abunde en su confesión y más se extienda será más digno de ser alabado. (3) Asimismo, quienes están obligados a traer ofrendas por pecado o por culpa, ya sean de carácter involuntario o deliberado, en el momento de ofrendarlas, no serán perdonados a menos que se arrepientan y se confiesen verbalmente, pues así está dicho: “Se confesó sobre aquello que pecó” (Levítico 5:5). Igualmente, todos los condenados a muerte o a azotes por decreto de un Tribunal, no quedan expiados sólo con la muerte o con los azotes, en tanto no se arrepientan y se confiesen. Asimismo, quien hiere a su prójimo o daña su propiedad, aunque le pague lo debido, no es perdonado hasta que se confiese y determine no volverlo a hacer nunca más, tal como versa: “De todos los pecados del hombre” (Números 5:6).

    Moral

    Séfer Hajaredim, 69a

    Es apropiado servir al Rey con objetos preciosos, de plata y oro. Y he aquí que el corazón es el órgano más preciado entre todos los doscientos cuarenta y ocho miembros del ser humano. Sobre esto se dijo: “Como plata selecta es la lengua del justo” (Proverbios 10:20). Asimismo, los dos ojos son como luminarias del firmamento. Los diez dedos frente a los diez Nombres Sagrados, las diez esferas. Los oídos son utensilios sagrados para escuchar las Palabras del Eterno, como está escrito: “Escuchad, y vivirán vuestras almas” (Isaías 55:3). Y el órgano genital que sirve para procrear a las huestes del Eterno, sus siervos, los que cumplen Su voluntad. ¡Cuánta santidad es necesaria, o algo más que santidad, pureza y candidez! El cerebro debemos ocuparlo con pensamientos puros, apegarse al Creador y para profundizar en Su Torá. El corazón, para regocijarse con Él y con Sus mandamientos, para apiadarse de los pobres y de los huérfanos. Que se sienta sosegado, sin tristeza ni ira. Los pies son los corredores que se apuran para cumplir la palabra del Rey. Paralelamente a los siete servidores se encuentran siete hebras del Tzitzit, y la octava representa a los corredores. El alma y el espíritu son representados por los Tefilín de la mano y de la cabeza. Para señalar creó los diez dedos, para atemorizar a la persona a que no peque. Es conveniente colocar la correa del Tefilín colgando frente a la cara como, son cadenas, mitad de hierro mitad de fuego. Por tanto, cuida los diez sirvientes, y lleva a cabo tu servicio con ellos. Cuando es necesario, ¿cuánto has de humillarte ante los hombres para que te presten dinero? ¿Cuántas súplicas? ¿Cuántas veces te despedirán sin nada? Humíllate ante el Eterno mil veces más, ya que todo depende de Su voluntad, Bendito Sea. ¿Cómo no te avergüenzas al pedir en tu plegaria tantas peticiones y tantos favores ante Él, Bendito Sea, cuando tu oración no va acompañada de súplicas y llantos?
