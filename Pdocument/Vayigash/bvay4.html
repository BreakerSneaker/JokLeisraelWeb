    (לג) ישב נא עבדך וגו'. לכל דבר אני מעולה ממנו, לגבורה ולמלחמה ולשמש: (א) ולא יכול יוסף להתאפק לכל הנצבים. לא היה יכול לסבול שיהיו מצרים נצבים עליו ושומעין שאחיו מתביישין בהודעו להם: (ב) וישמע בית פרעה. ביתו של פרעה, כלומר עבדיו ובני ביתו, ואין זה לשון בית ממש, אלא כמו בית ישראל, בית יהודה, מישנ"דה בלע"ז: (ג) נבהלו מפניו. מפני הבושה: (ד) גשו נא אלי. ראה אותם נסוגים לאחור, אמר, עכשיו אחי נכלמים, קרא להם בלשון רכה ותחנונים והראה להם שהוא מהול (ב"ר צג, ח.):

    ב בֶּן־אָדָ֗ם שִֹ֤ים פָּנֶ֙יךָ֙ אֶל־גּוֹג֙ אֶ֣רֶץ הַמָּג֔וֹג נְשִֹ֕יא רֹ֖אשׁ מֶ֣שֶׁךְ וְתֻבָ֑ל וְהִנָּבֵ֖א עָלָֽיו: בַּר אָדָם קַבֵּיל נְבוּאָה עַל גּוֹג אַרְעָא מָגוֹג רַב רֵישׁ מֶשֶׁךְ וְתוּבָל וְאִתְנַבֵּי עֲלוֹהִי: ג וְאָ֣מַרְתָּ֔ כֹּ֥ה אָמַ֖ר אֲדֹנָ֣י יֱהֹוִ֑ה הִנְנִ֤י אֵלֶ֙יךָ֙ גּ֔וֹג נְשִֹ֕יא רֹ֖אשׁ מֶ֥שֶׁךְ וְתֻבָֽל: וְתֵימַר כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא שָׁלַח רוּגְזִי עֲלָךְ גּוֹג רַב רֵישׁ מֶשֶׁךְ וְתוּבָל: ד וְשׁ֣וֹבַבְתִּ֔יךָ וְנָתַתִּ֥י חַחִ֖ים בִּלְחָיֶ֑יךָ וְהוֹצֵאתִי֩ אוֹתְךָ֙ וְאֶת־כָּל־חֵילֶ֜ךָ סוּסִ֣ים וּפָרָשִׁ֗ים לְבֻשֵׁ֤י מִכְלוֹל֙ כֻּלָּ֔ם קָהָ֥ל רָב֙ צִנָּ֣ה וּמָגֵ֔ן תֹּפְשֵֹ֥י חֲרָב֖וֹת כֻּלָּֽם: וַאֲשַׁדְלִינָךְ וְאֶתֵּן שֵׁירִין עַל לִיסָתָךְ וְאַפֵּיק יָתָךְ וְיַת כָּל מַשִׁרְיָתָךְ סּוּסְוָן וּפָרָשִׁין לְבוּשֵׁי גְּמִיר כּוּלְהוֹן מַשִּׁרְיָן סַגִּיאִין דִּי מְזַיְנִין בַּעֲגִילִין וּתְרֵיסִין אֲחֲדֵי סַיְפִין כּוּלְהוֹן: ה פָּרַ֛ס כּ֥וּשׁ וּפ֖וּט אִתָּ֑ם כֻּלָּ֖ם מָגֵ֥ן וְכוֹבָֽע: פַּרְסָאֵי כּוּשָׁאֵי וּפוּטָאֵי עִמְּהוֹן כּוּלְהוֹן מְזַיְנִין בַּעֲגִילִין וְקוּלְסִין: ו גֹּמֶר וְכָל־אֲגַפֶּ֔יהָ֚ בֵּית תּֽוֹגַרְמָ֔ה יַרְכְּתֵ֥י צָפ֖וֹן וְאֶת־כָּל־אֲגַפָּ֑יו עַמִּ֥ים רַבִּ֖ים אִתָּֽךְ: גּוֹמֶר וְכָל מַשְׁרִיָּיתָהָא מְדִינַת גַּרְמַמְיָא סִפֵּי צִפּוּנָא וְיַת כָּל מַשִּׁרְיָתְהוֹן עַמְמִין סַגִּיאִין דִּי עִמָּךְ: ז הִכֹּן֙ וְהָכֵ֣ן לְךָ֔ אַתָּ֕ה וְכָל־קְהָלֶ֖ךָ הַנִּקְהָלִ֣ים עָלֶ֑יךָ וְהָיִ֥יתָ לָהֶ֖ם לְמִשְׁמָֽר: אִזְדַּרַז וְאַתְקֵין לָךְ אַתְּ וְכָל מַשִׁרְיָתָךְ דְּאִתְכְּנִישׁוּ עֲלָךְ וּתְהֵא לְהוֹן לְמַטָּרָא:

    משלי י

    כ כֶּ֣סֶף ֖נִבְחָר לְשׁ֣וֹן צַדִּ֑יק לֵ֖ב רְשָׁעִ֣ים כִּמְעָֽט: סָאמָא גַבְיָא לִשְׁנֵיהּ דְּצַדִיקָא וְלִבָּא דְרַשִּׁיעֵי מַחֲתָא: כא שִׂפְתֵ֣י ֖צַדִּיק יִרְע֣וּ רַבִּ֑ים ֜וֶֽאֱוִילִ֗ים בַּחֲסַר־לֵ֥ב יָמֽוּתוּ: שִׂפְוָתָא דְצַדִיקֵי רָעְיָן סַגִיאִי וְשַׁטְיֵי בַּחֲסִירוּת רַעְיָנָא יְמוּתוּן: כב בִּרְכַּ֣ת ֖יְהֹוָה הִ֣יא תַעֲשִׁ֑יר וְלֹֽא־יוֹסִ֖ף עֶ֣צֶב עִמָּֽהּ: בִּרְכְּתָא דֶאֱלָהָא הִיא מְעַתְּרָא וְלָא נוֹסֵף כֵּיבָא עִמָהּ: כג כִּשְׂח֣וֹק ֖לִכְסִיל עֲשׂ֣וֹת זִמָּ֑ה ֜וְחָכְמָ֗ה לְאִ֣ישׁ תְּבוּנָֽה: כַּד גָּחַךְ סַכְלָא עָבַד עֲבִידְתָא וְחָכְמְתָא לְגַבְרָא דְמִתְבַּיֵן: כד מְגוֹרַ֣ת ֖רָשָׁע הִ֣יא תְבוֹאֶ֑נּוּ וְתַאֲוַ֖ת צַדִּיקִ֣ים יִתֵּֽן: חֲרִיצוּתָא דְרַשִּׁיעָא הִיא אָתְיָא לֵיהּ וְרִגְגְתָא דְצַדִיקַיָא תִתְיְהֵב לְהוֹן: כה כַּעֲב֣וֹר ֖סוּפָה וְאֵ֣ין רָשָׁ֑ע ֜וְצַדִּ֗יק יְס֣וֹד עוֹלָֽם: הֵיךְ דְּיֶעְבַּר עַלְעוּלָא הֵיכְנָא נֶעְבַּר רַשִּׁיעָא וְצַדִיקָא שִׁתְאֲסָיָא בְּעָלְמָא:

    רש"י

    (כ) כֶּסֶף נִבְחַר לְשׁוֹן צַדִּיק. שֶׁיּוֹדֵעַ לְהוֹכִיחַ: לֵב רְשָׁעִים כִּמְעַט. שֶׁאֵינוֹ שׁוֹמֵעַ לְתּוֹכַחַת צַדִּיק, וּמִדְרָשׁ רַבִּי תַנְחוּמָא אוֹמֶר (כִּי תִּשָּׂא ו): עַל עִדּוֹ הַנָּבִיא נֶאֱמַר, שֶׁקָּרָא עַל הַמִּזְבֵּחַ בְּבֵית אֵל (מְלָכִים א' י"ג ב'), וְיָרָבְעָם אַף עַל פִּי שֶׁיָבְשָׁה יָדוֹ, לֹא קִבֵּל תּוֹכֵחָה, שֶׁנֶּאֱמַר (מְלָכִים שָׁם פָּסוּק ו): חָל נָא אֶת פְּנֵי ה' אֱלֹהֶיךָ וגו', וְלֹא אֱלֹהַי, וְאוֹמֵר: וְתֵשֵׁב יָדוֹ כְּבָרִאשׁוֹנָה, עוֹמֵד וּמַקְטִיר לַעֲבוֹדַת גִּלוּלִים וְאַף בַּסּוֹף כֵּן (שָׁם לג): (כא) שִׂפְתֵי צַדִּיק יִרְעוּ רַבִּים וגו'. שֶׁרָבִים אוֹכְלִים בִּזְכוּתוֹ וּבִתְפִלָּתוֹ: (כב) בִּרְכַּת ה' הִיא תַּעֲשִׁיר, וְלֹא יוֹסִיף עֶצֶב עִמָּהּ. אֵין צָרִיךְ לְהִתְיַיגֵעַ לְהַעֲשִׁיר בּוֹ, דַּי בַּבְּרָכָה מָה שֶׁהוּא מְבַּרְכוֹ: (כג) כִּשְׂחֹק לִכְסִיל עֲשׂוֹת זִמָּה. עַצַת חָטָאִים: וְחָכְמָה. כְּשְׂחֹק: לְאִישׁ תְּבוּנָה. כְּלוֹמַר קַלָּה הִיא בְּעֵינָיו לַעֲשׂוֹת: (כד) מְגוֹרַת רָשָׁע. מָה שֶׁהוּא יָרֵא, יָבֹא לוֹ. דּוֹר הַפָּלָגָה יַרְאוּ וְאָמְרוּ (בֶּרֶאשִׁית י) פֶּן נָפוֹץ, וְסוֹפָן כְּתִיב וַיָפֶץ ה' אוֹתָם מִשָּׁם (שָׂם): וְתַאֲוַת צַדִּיקִים יִתֵּן. מִי שֶׁבְּיָדוֹ לִיתְּנָהּ: (כה) כַּעֲבֹר סוּפָה וְאֵין רָשָׁע. פִּתְאוֹם בָּא עֶבְרַת רוּחַ סְעָרָה, וְנִכְרָת הָרָשָׁע מִמְּקוֹמוֹ:

    א שְׁנַיִם אוֹחֲזִין בְּטַלִּית, זֶה אוֹמֵר אֲנִי מְצָאתִיהָ וְזֶה אוֹמֵר אֲנִי מְצָאתִיהָ, זֶה אוֹמֵר כֻּלָּהּ שֶׁלִּי וְזֶה אוֹמֵר כֻּלָּהּ שֶׁלִּי, זֶה יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוֹת מֵחֶצְיָהּ, וְזֶה יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוֹת מֵחֶצְיָהּ, וְיַחֲלֹקוּ. זֶה אוֹמֵר כֻּלָּהּ שֶׁלִּי וְזֶה אוֹמֵר חֶצְיָהּ שֶׁלִּי, הָאוֹמֵר כֻּלָּהּ שֶׁלִּי, יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוֹת מִשְּׁלשָׁה חֲלָקִים, וְהָאוֹמֵר חֶצְיָהּ שֶׁלִּי, יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוֹת מֵרְבִיעַ. זֶה נוֹטֵל שְׁלשָׁה חֲלָקִים, וְזֶה נוֹטֵל רְבִיעַ: ב הָיוּ שְׁנַיִם רוֹכְבִין עַל גַּבֵּי בְהֵמָה, אוֹ שֶׁהָיָה אֶחָד רוֹכֵב וְאֶחָד מַנְהִיג, זֶה אוֹמֵר כֻּלָּהּ שֶׁלִּי, וְזֶה אוֹמֵר כֻּלָּהּ שֶׁלִּי, זֶה יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוֹת מֵחֶצְיָהּ, וְזֶה יִשָּׁבַע שֶׁאֵין לוֹ בָהּ פָּחוֹת מֵחֶצְיָהּ, וְיַחֲלֹקוּ. בִּזְמַן שֶׁהֵם מוֹדִים אוֹ שֶׁיֵּשׁ לָהֶן עֵדִים, חוֹלְקִים בְּלֹא שְׁבוּעָה: ג הָיָה רוֹכֵב עַל גַּבֵּי בְהֵמָה וְרָאָה אֶת הַמְּצִיאָה, וְאָמַר לַחֲבֵרוֹ תְּנֶהָ לִי, נְטָלָהּ וְאָמַר אֲנִי זָכִיתִי בָהּ, זָכָה בָהּ. אִם מִשֶּׁנְּתָנָהּ לוֹ אָמַר אֲנִי זָכִיתִי בָהּ תְּחִלָּה, לֹא אָמַר כְּלוּם: ד רָאָה אֶת הַמְּצִיאָה וְנָפַל עָלֶיהָ, וּבָא אַחֵר וְהֶחֱזִיק בָּהּ, זֶה שֶׁהֶחֱזִיק בָּהּ זָכָה בָהּ. רָאָה אוֹתָן רָצִין אַחַר מְצִיאָה, אַחַר צְבִי שָׁבוּר, אַחַר גּוֹזָלוֹת שֶׁלֹּא פָרְחוּ, וְאָמַר זָכְתָה לִי שָׂדִי, זָכְתָה לוֹ. הָיָה צְבִי רָץ כְּדַרְכּוֹ, אוֹ שֶׁהָיוּ גוֹזָלוֹת מַפְרִיחִין, וְאָמַר זָכְתָה לִי שָׂדִי, לֹא אָמַר כְּלוּם: ה מְצִיאַת בְּנוֹ וּבִתּוֹ הַקְּטַנִּים, מְצִיאַת עַבְדּוֹ וְשִׁפְחָתוֹ הַכְּנַעֲנִים, מְצִיאַת אִשְׁתּוֹ, הֲרֵי אֵלּוּ שֶׁלּוֹ. מְצִיאַת בְּנוֹ וּבִתּוֹ הַגְּדוֹלִים, מְצִיאַת עַבְדּוֹ וְשִׁפְחָתוֹ הָעִבְרִים, מְצִיאַת אִשְׁתּוֹ שֶׁגֵּרְשָׁהּ, אַף עַל פִּי שֶׁלֹּא נָתַן כְּתֻבָּתָהּ, הֲרֵי אֵלּוּ שֶׁלָּהֶן: ו מָצָא שְׁטָרֵי חוֹב, אִם יֵשׁ בָּהֶן אַחֲרָיוּת נְכָסִים, לֹא יַחֲזִיר, שֶׁבֵּית דִּין נִפְרָעִין מֵהֶן, אֵין בָּהֶן אַחֲרָיוּת נְכָסִים, יַחֲזִיר, שֶׁאֵין בֵּית דִּין נִפְרָעִין מֵהֶן, דִּבְרֵי רַבִּי מֵאִיר. וַחֲכָמִים אוֹמְרִים, בֵּין כָּךְ וּבֵין כָּךְ לֹא יַחֲזִיר, מִפְּנֵי שֶׁבֵּית דִּין נִפְרָעִין מֵהֶן: ז מָצָא גִטֵי נָשִׁים, וְשִׁחְרוּרֵי עֲבָדִים, דִּיָּתִיקִי, מַתָּנָה וְשׁוֹבָרִים, הֲרֵי זֶה לֹא יַחֲזִיר, שֶׁאֲנִי אוֹמֵר כְּתוּבִים הָיוּ וְנִמְלַךְ עֲלֵיהֶם שֶׁלֹּא לִתְּנֵם: ח מָצָא אִגְּרוֹת שׁוּם וְאִגְּרוֹת מָזוֹן, שְׁטָרֵי חֲלִיצָה וּמֵאוּנִין, וּשְׁטָרֵי בֵרוּרִין, וְכָל מַעֲשֵׂה בֵית דִּין, הֲרֵי זֶה יַחֲזִיר. מָצָא בַחֲפִיסָה אוֹ בִדְלֻסְקְמָא, תַּכְרִיךְ שֶׁל שְׁטָרוֹת, אוֹ אֲגֻדָּה שֶׁל שְׁטָרוֹת, הֲרֵי זֶה יַחֲזִיר. וְכַמָּה אֲגֻדָּה שֶׁל שְׁטָרוֹת. שְׁלשָׁה קְשׁוּרִין זֶה בָזֶה. רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר, אֶחָד הַלֹּוֶה מִשְּׁלשָׁה, יַחֲזִיר לַלֹּוֶה, שְׁלשָׁה הַלֹּוִין מֵאֶחָד, יַחֲזִיר לַמַּלְוֶה. מָצָא שְׁטָר בֵּין שְׁטָרוֹתָיו וְאֵינוֹ יוֹדֵעַ מַה טִיבוֹ, יְהֵא מֻנָּח עַד שֶׁיָּבוֹא אֵלִיָּהוּ. אִם יֵשׁ עִמָּהֶן סִמְפּוֹנוֹת, יַעֲשֶׂה מַה שֶּׁבַּסִּמְפּוֹנוֹת:

    א שנים אוחזין בטלית. בגמרא מוקי למתניתין כגון שאחד מהם תופס בחוטין שבשפת הבגד מצד זה, והאחד תופס בחוטין שבשפת הבגד מצד זה. אבל אם הם אדוקים בבגד עצמו, זה נוטל עד מקום שידו מגעת וזה נוטל עד מקום שידו מגעת, והשאר חולקים בשוה, ובשבועה: זה אומר כולה שלי. אני קניתיה ולי מכרה המוכר ולא לך. והמוכר מכר לאחד מהם ולקח המעות משניהם, מאחד מדעתו ומאחד בעל כרחו, ואינו יודע מהי מדעתו ומהי בעל כרחיה. דאילו ידע והיה אומר לזה מכרתי, היה כאן עד אחד, והיה שכנגדו חייב שבועה דאורייתא להכחיש העד. עכשיו שאינו יודע, שניהם נשבעים שבועה זו האמורה במתניתין. ובדין הוא שיהיו חולקים בלא שבועה, אלא שתקנו חכמים שלא יטול שום אחד מהם אלא בשבועה, כדי שלא יהא כל אחד הולך ותוקף בטליתו של חבירו ואומר שלי היא. ואיצטריך תנא לאשמועינן בזה אומר אני מצאתיה דהיינו במציאה, ובזה אומר כולה שלי דהיינו מקח וממכר. אי תנא מציאה, הוה אמינא מציאה הוא דרמו רבנן שבועה עליה משום דמורה התירא לאחוז בה שלא כדין, דאמר חבראי לאו מידי חסר בה איזיל ואתפוס ואפלוג בהדיה, אבל מקח וממכר דאי לא הוה צריך לה לא הוה מהדר אבתרה למזבנה, וזה שבא לחלוק עמו וליתן חצי דמיה שלא כדין מחסרו וליכא למימר דמורי התירא, אימא לא רמו רבנן שבועה עליה. ואי אשמועינן במקח וממכר, הוה אמינא מקח וממכר הוא דרמו רבנן שבועה עליה משום דמורי התירא ואמר חבראי דמי קא יהיב ואנא דמי קא יהיבנא, השתא דצריכה לדידי אשקליה אנא, וחבראי לטרח וליזיל ולזבין אחריתי, אבל מציאה דליכא למימר הכי, אימא לא. צריכא: ישבע שאין לו בה פחות מחציה. ואינו נשבע שכולה שלו כדקא טעין מעיקרא, דהא לא יהבי ליה כולה. ואי משתבע שחציה שלו כדקא יהבי ליה, הוה מרע ליה לדבוריה קמא דאמר כולה שלי. הלכך ישבע שאין לו בה פחות מחציה, דמשמע הכי, כולה שלי כדקא אמינא מעיקרא, ולדבריכם שאין אתם מאמינים לי בכולה, שבועה שיש לי בה ואין לי בה פחות מחציה: ב היו שנים רוכבים על גבי בהמה. הא קמ"ל דרוכב קני, ואע"פ שאינו מנהיג שלא זזה הבהמה ממקומה: או שהיה אחד רוכב ואחד מנהיג. בזמן שרוכב מנענע ברגליו כי היכי דתיזיל הבהמה מחמתיה, הוא דשוים רוכב ומנהיג. ואם לא היה אלא רוכב בלבד, מנהיג קנה, רוכב לא קנה. ואם הודו או באו עדים אפילו לאחר שנפסק הדין עליהם שיחלוקו בשבועה, חולקים שלא בשבועה: ג אמר לחבירו תנה לי וכו'. אבל אמר זכה לי בה, קנה הרוכב, ואין המגביה יכול לומר אני זכיתי בה: לא אמר כלום. שקנאה זה שמשכה מיד חברו, וכל זמן שהיתה ביד המגביה הפקר היתה: ד זה שהחזיק בה זכה בה. ודוקא שהיתה המציאה ברשות הרבים, שאין ארבע אמות של אדם קונות לו ברה"ר לפיכך המחזיק בה זכה. אבל בסימטא שהיא שביל של יחיד, או בצדי רשות הרבים שאין רבים דוחקים שם, ארבע אמות של אדם קונות לו כל מציאה וכל הפקר שסמוך לו בארבע אמות, ואין אחר רשאי לתפסו. ותקינו ליה רבנן כי היכי דלא ליתו לאינצויי: אחר צבי שבור. שאינו יכול לרוץ ומשתמר בתוך השדה אם לא יטלוהו אחרים, והוי כמציאה: זכתה לו. שדהו. והוא שעומד בצד שדהו, וכגון שיכול לרוץ אחריהן ומגיען קודם שיצאו משדהו: ה הקטנים. כל שסמוך על שלחן אביו אפילו הוא גדול קרי ליה קטן ומציאתו לאביו משום איבה. והבת בין קטנה בין נערה מציאתה לאביה, דהתורה זיכתה כל שבח נעוריה לאביה: עבדו ושפחתו הכנענים. שהרי גופן קנוי לו. כדכתיב (ויקרא ב"ה) והתנחלתם אותם: מציאת אשתו. רבנן תקינו ליה משום איבה: ו אחריות נכסים. שעבוד קרקעות שיגבה מהם: לא יחזיר. דחיישינן לפרעון ולקנוניא. שמא שטר פרוע הוא ומן הלוה נפל, והא דקא מודה לא פרעתי, עצה של רמאות היא ביניהם לטרוף את הלקוחות שלקחו ממנו קרקע שלא באחריות, ויחלקוה ביניהם: בין כך ובין כך לא יחזיר. דשטר שאין בו אחריות ממשעבדי נמי גבי, דאחריות טעות סופר הוא, וחיישינן לפרעון ולקנוניא. והלכה כחכמים. ודוקא בשטר שאין נזכר בו אחריות אמרו חכמים דטעות סופר הוא וגבי ממשעבדי כאילו היה האחריות כתוב בו. אבל אם פירש בשטר שאינו רוצה לקבל עליו אחריות, מודים חכמים דיחזיר, דהשתא ליכא למיחש לקנוניא: ז דייתיקי. צוואת שכיב מרע. דא תהא למיקם ולהיות: ושוברים. שעושה מלוה ללוה שנפרע ממנו שטר חובו: ח אגרות שום. ששמו בית דין נכסי לוה למלוה בחובו: ואגרות מזון. שקבל עליו לזון את בת אשתו. פירוש אחר, שימכרו מקרקע הבעל למזון האשה והבנות: ומיאונין. שכותבים בשטר בפנינו מיאנה פלונית בפלוני בעלה. ובקטנה שהשיאוה אמה ואחיה שאינה צריכה גט: שטרי בירורין. זה בורר לו אחד וזה בורר לו אחד שידונו להם: מצא. שטרות: בחפיסה. חמת של עור קטנה: בדלוסקמא. כלי עור שהזקנים מצניעים בה כלי תשמישם שלא יצטיכו לחפש אחריהם: תכריך של שטרות. שלשה שטרות או יותר כרוכין זה בזה: אגודה. מושכבים זה על זה, ארכו של זה על ארכו של זה: הרי זה יחזיר. דדבר שיש בו סימן הוא, שהכלי סימן כשיאמרו הבעלים בכלי פלוני מצאת אותם. וכן תכריך ואגודה סימן הוא: אחד הלוה משלשה. אם שלשה שטרות של לוה אחד הן שלוה משלשה בני אדם. יחזירם המוצא ללוה, דודאי מידו נפלו, שאם מידם נפלו מי קבצם למקום אחד. ודוקא שהשטרות מקוימים בבית דין. אבל אם אינם מקוימים חיישינן דלמא לקיימינהו הוליכום שלשה המלוים אצל סופר הדיינים ונפלו מיד הסופר. מיד הסופר נפלו ולא לוו מעולם, ולפיכך לא יחזיר: מצא שטר בין שטרותיו ואינו יודע מה טיבו. אצלו. אם הלוה הפקידו אצלו או המלוה, או שמא מקצתו פרוע ומסרוהו לו להיות שליש ביניהם: יהא מונח. בידו, ולא יחזיר לא     בבא מציעא דף כ"ט ע"ב

    תָּנוּ רַבָּנָן הַשּׁוֹאֵל סֵפֶר תּוֹרָה מֵחֲבֵרוֹ הֲרֵי זֶה לֹא יַשְׁאִילֶנּוּ לְאַחֵר פּוֹתְחוֹ וְקוֹרֵא בוֹ וּבִלְבַד שֶׁלֹּא יִלְמוֹד בּוֹ בַּתְּחִלָּה וְלֹא יִקְרַא אַחֵר עִמּוֹ. וְכֵן הַמַּפְקִיד סֵפֶר אֵצֶל חֲבֵרוֹ גּוֹלְלוֹ כָּל שְׁנֵים עָשָׂר חֹדֶשׁ פּוֹתְחוֹ וְקוֹרֵא בוֹ אִם בִּשְׁבִילוֹ פְּתָחוֹ אָסוּר. סוּמְכוֹס אוֹמֵר בֶּחָדָשׁ שְׁלשִׁים יוֹם בְּיָשָׁן שְׁנֵים עָשָׂר חֹדֶשׁ. רַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר אֶחָד זֶה וְאֶחָד זֶה שְׁנֵים עָשָׂר חֹדֶשׁ. אָמַר מָר הַשּׁוֹאֵל סֵפֶר תּוֹרָה מֵחֲבֵרוֹ הֲרֵי זֶה לֹא יַשְׁאִילֶנּוּ לְאַחֵר מָאי אִרְיָא סֵפֶר תּוֹרָה אֲפִילוּ כָּל מִילֵי נָמֵי דְּאָמַר רַבִּי שִׁמְעוֹן בֶּן לָקִישׁ כָּאן שָׁנָה רַבִּי אֵין הַשּׁוֹאֵל רַשָּׁאי לְהַשְׁאִיל וְאֵין הַשׂוֹכֵר רַשָּׁאי לְהַשְׂכִּיר. סֵפֶר תּוֹרָה אִיצְטְרִיכָא לֵיהּ מַהוּ דְתֵימָא נִיחָא לֵיהּ לְאִינִישׁ דְתַעֲבִיד מִצְוָה בְּמָמוֹנֵיהּ קָא מַשְׁמַע לָן. פּוֹתְחוֹ וְקוֹרֵא בוֹ פְּשִׁיטָא וְאֶלָּא לְמָאי שַׁיְילוּהָ מִינֵיהּ סֵיפָא אִיצְטְרִיכָא לֵיהּ וּבִלְבַד שֶׁלֹּא יִלְמוֹד בּוֹ בַתְּחִלָּה. וְכֵן הַמַּפְקִיד סֵפֶר תּוֹרָה אֵצֶל חֲבֵירוֹ גוֹלְלוֹ כָּל שְׁנֵים עָשָׂר חֹדֶשׁ פּוֹתְחוֹ וְקוֹרֵא בוֹ מָאי עֲבִידְתֵּיהּ גַבָּהּ. וְתוּ אִם בִּשְׁבִילוֹ פּוֹתְחוֹ אָסוּר הָא אֲמַרְתְּ פּוֹתְחוֹ וְקוֹרֵא בוֹ. הָכִי קָאָמַר אִם כְּשֶׁהוּא גוֹלְלוֹ פּוֹתְחוֹ וְקוֹרֵא בוֹ מוּתָּר וְאִם בִּשְׁבִילוֹ פְּתָחוֹ אָסוּר. סוּמְכוֹס אוֹמֵר בֶּחָדָשׁ שְׁלשִׁים יוֹם בְּיָשָׁן שְׁנֵים עָשָׂר חֹדֶשׁ. רַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר אֶחָד זֶה וְאֶחָד זֶה שְׁנֵים עָשָׂר חֹדֶשׁ. רַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב הַיְינוּ תַּנָּא קַמָּא אֶלָּא אֵימָא רַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב אוֹמֵר אֶחָד זֶה וְאֶחָד זֶה שְׁלשִׁים יוֹם אֲבָל לֹא יִלְמוֹד בּוֹ בַּתְּחִלָּה וְלֹא יִקְרָא אַחֵר עִמּוֹ וּרְמִינְהִי לֹא יִקְרָא פָּרָשָׁה וְיִשְׁנֶה וְלֹא יִקְרָא בוֹ פָּרָשָׁה וִיתַרְגֵּם וְלֹא יִפְתַּח בּוֹ יוֹתֵר מִשְּׁלשָׁה דַפִּין וְלֹא יִקְרְאוּ בוֹ שְׁלשָׁה בְנֵי אָדָם בְּכֶרֶךְ אֶחָד הָא שְׁנַיִם קוֹרִין אָמַר אַבַּיֵי לֹא קַשְׁיָא כַּאן בְּעִנְיָן אֶחָד כַּאן בִּשְׁנֵי עִנְיָנִים:

    תרגום הגמרא

    שנו רבותינו בברייתא: השואל (לוקח בהשאלה) ספר תורה מחברו הרי זה לא ישאילנו לאדם אחר. מותר לו לפותחו ולקרוא בו אבל רק מה שלמד כבר פעם ולא פרשה שאיננו יודע אותה כלל, זאת מחשש שבשעה שהוא מעיין בלימודו הוא עלול לקלקל את הספר. ולא יקרא אדם אחר עמו. וכן המפקיד ספר תורה אצל חברו גולל השומר את הספר פעם בשנים עשר חודשים כדי שלא יתעפש ויתקלקל ולשם שמירת הספר מותר לו לפותחו ואף לקרוא בו, אך אם פתח את הספר לצורך עצמו כדי ללמוד בו- אסור. סומכוס אומר: בספר תורה חדש- פותח אותו וגוללו כל שלושים יום, ובספר ישן- כל שנים עשר חודש, זאת משום שהחדש ממהר להתעפש יותר מהישן. רבי אליעזר בן יעקב אומר: אחד זה ואחד זה (הן חדש והן ישן) יפתח ויגלול אותו כל שנים עשר חודש. למדנו: "השואל בפר תורה מחברו הרי זה ח\לא ישאילנו לאדם אחר", ושואלים בגמרא: מדוע נאמרה ההלכה על ספר תורה? והרי כך הדין גם בשאר דברים, שהרי אמר רבי שמעון בן לקיש (על דברי המשנה בגיטין "השולח גט... אם אמר לו טול (קח) חפץ פלוני הימנה (ממנה) לא ישלחנו ביד אחר"): כאן, במשנה זו לימד אותנו רבי שאין השואל רשאי להשאיל ואין השוכר רשאי להשכיר?! ומתרצים: בספר תורה היה צורך לומר זאת במפורש, משום שהיינו טועים לומר: נוח לו לאדם שתיעשה מצוה בממונו ועל כן אין הוא מתנגד שיפתחו את ספרו וילמדו בו- משמיעה לנו הברייתא שאסור להשאילו לאחר. "פותחו וקורא בו". שואלים בגמרא: פשיטא! ברור ומובן שמותר לו לקרוא בו, שהרי בשואל מדובר ולשם מה שאל את הספר? ומתרצים: "פותחו וקורא בו" נאמר בשל דברי הסיפא בהמשך: "ובלבד שלא ילמוד בו בתחילה" כלומר שלא ילמד בו פרשה שמעולם לא למד אותה. "וכן המפקיד ספר תורה אצל חבירו גוללו כל שנים עשר חודש, פותחו וקורא בו". שואלים בגמרא: מה מעשיו של השומר אצל הספר שהופקד אצלו, כלומר מדוע יהיה מותר לו לפותחו ולקרוא בו? ועוד קשה: הרי מיד אחרי זה נאמר "אם בשבילו פתחו אסור" ואיך זה מתיישב עם "פותחו וקורא בו"? ומתרצים: כך אמרה הברייתא: אם כשהוא גולל את הספר (כדי לאורר אותו) פותחו וקורא בו- מותר, ואם לצורך עצמו פתח את הספר- אסור. "סומכוס אומר: בחדש- גוללו כל שלושים יום, בישן- כל שנים עשר חודש. רבי אליעזר בן יעקב אומר: אחד זה ואחד זה – כל שנים עשר חודש". שואלים: שיטת רבי אליעזר בן יעקב היא היא שיטת תנא קמא (התנא הראשון) האומר בתחילת הברייתא, מבלי לחלק בין ספר ישן לחדש, "גוללו כל שנים עשר חודש"? ומתרצים: אכן כך צריך לגרוס בברייתא: "רבי אליעזר בן יעקב אומר: אחד זה ואחד זה (בין ישן בין חדש, גוללו) כל שלושים יום". "אבל לא ילמוד בו בתחילה ולא יקרא אחר עמו". מקשים על כך: הרי למדנו: "לא יקרא פרשה וישנה (ויחזור עליה בשנית) ולא יקרא בו פרשה ויתרגם, ולא יפתח בו יותר משלושה דפים (כלומר שלושה גליונות כי אז הוא עלול להיקרע) ולא יקראו בו שלושה בני אדם בכרך אחד" ומשמע מכאן ששנים יחד מותר להם לקרוא בו? אמר אביי: אין זו קושיא- כאן שנאמר "ולא יקרא אחר עמו", מדובר כששניהם מעיינים באותו דף ואז עלול להיות שכל אחד מהם ימשוך את הספר שיהיה קרוב אליו והספר יקרע, ואילו כאן- שנאמר "ולא יקראו בו שלושה" ומשמע שלשנים מותר, מדובר כאשר כל אחד מהם קורא בדף אחר.

    רש"י

    גוללו בכל שנים עשר חדש פעם אחת: אם בשבילו פתחו כולה מפרש לקמיה: מאי אריא ספר שהוא נוח להתקלקל בטשטוש וקריע': כאן שנה רבי במס' גיטין תנן השולח גט בארץ ישראל הרי זה משלחו ביד אחר ואם אמר לו טול חפץ פלוני הימנה לא ישלחנו ביד אחר שאין רצונו שיהא פקדונו ביד אחר ואמר ר"ל עלה כאן שנה רבי במשנה זו למדנו רבי שסתם המשניות אין השואל רשאי וכו' דבכולהו שייך למימר אין רצונו שיהא פקדונו ביד אחר: מאי עבידתיה גבה קס"ד לצורכו קאמר ולא לצורך ספר תורה דהא לצורך ס"ת תנא ליה גוללו כל שנים עשר חדש להכי פרכינן מאי עבידתי' דהאי גבי ספר תורה לקרות בו לצורך עצמו: אם כשהוא גוללו להנאת ספר תורה פתחו וקרא בו מותר: בחדש שלשים יום החדש ממהר להתעפש מן הישן: אימא אחד זה ואחד זה שלשים יום מתניתין נמי דקתני קורא בהן אחת לשלשים יום ולא מפליג בין ישן לחדש רבי אליעזר בן יעקב היא: בענין אחד בפרשה אחת אין קורין: בשני ענינים זה בדף זה וזה בדף זה קורין דלא אתי לשמוטי מהדדי:

    Rashí

    33. יֵשֶׁב נָא עַבְדְּךָ וגו' / QUE TU SIERVO SE QUEDE, ETC. Te conviene quedarte conmigo, pues en todo yo soy superior a él: en fuerza, en capacidad para hacer la guerra y para servir. 45 1. וסֵף לְהִתְאַפֵּק לְכל הַנִּצַּבִים י וְלא יָכל / YOSEF NO PUDO CONTENERSE MÁS A CAUSA DE TODOS LOS QUE ESTABAN PARADOS [JUNTO A ÉL]. No podía soportar que hubiera mitzrim en su presencia que escucharan a sus hermanos avergonzarse cuando él se diera a conocer a ellos. 2. בֵּית פַּרְעֹה וַיִּשְׁמַע / Y OYÓ LA CASA DEL FARAÓN. La frase בֵּית פַּרְעֹה significa “la casa del Faraón”, es decir, sus servidores y los miembros de su casa. En este caso, esta expresión no se refiere a una “casa” en sentido físico, sino que es similar en significado a las expresiones: “la Casa de Israel”, “la Casa de Yehudá”. Significa lo mismo que mesnede en francés antiguo. 3.נִבְהֲלוּ מִפָּנָיו / SE HABÍAN TURBADO [PROFUNDAMENTE] DELANTE DE ÉL. A causa de la vergüenza que sentían por haberlo vendido. 4. גְּשׁוּ נָא אֵלַי/ POR FAVOR, ACÉRQUENSE A MÍ. Yosef vio que retrocedían a causa de su turbación, y se dijo: “Ahora mis hermanos sienten vergüenza”. Entonces los llamó con palabras suaves y expresiones conciliadoras y, para probarles que en efecto era él, les mostró que estaba circuncidado.

    Mishná

    Tratado Babá-Metsiá, Capítulo 1

    (1) Dos agarran un Talit. Uno dice: “Yo lo encontré” y el otro dice: “Yo lo encontré”. Uno dice: “Es todo mío” y el otro dice: “Es todo mío”. Uno jura que posee no menos de la mitad y el otro jura que le pertenece no menos de la mitad, lo dividirán. Uno dice: “Es todo mío”, y el otro dice: “La mitad es mío”. El que dijo “Todo es mío” deberá jurar que no posee menos de tres cuartas partes, y el que dijo “La mitad es mío” jurará que no posee menos de una cuarta parte. Finalmente uno se quedará con tres partes y el otro con la otra cuarta parte. (2) Dos montaban a la vez un mismo animal, o bien, uno lo montaba y el otro lo guiaba. Si uno dice: “Todo es mío” y el otro dice: “Todo es mío”, entonces, jurará uno que no le pertenece menos de la mitad, y el segundo jurará que no posee menos de la mitad y se lo repartirán. Si ambos concuerdan lo mismo o si hubo presencia de testigos, lo repartirán sin que haya necesidad de jurar. (3) Si un jinete cabalga sobre un animal y encuentra un objeto, y le dice a otro individuo: “Alcánzamelo”, y lo levanta y dice (el que alzó el objeto): “Lo tomé para mí”, le pertenece. Si después de dárselo dice: “Lo tomé para mí”, no vale lo que dijo. (4) Si vio un objeto y se abalanzó sobre él, y otro se le adelantó y lo agarró, aquel que lo asió se lo queda. Si alguien ve a ciertas personas corriendo para asir algo, como una gacela con una pata rota, o polluelos que no pueden volar, y dice el propietario de aquel campo: “Mi campo me otorgó la propiedad”, le pertenece. No obstante, si la gacela corría libremente o los polluelos volaban y dice: “Mi campo me otorgó la propiedad”, no le sirve lo que dijo. (5) Un objeto hallado por un hijo o una hija, menores de edad, o por un siervo o una sierva, quenaanitas, o por su esposa, le pertenece. Un objeto hallado por un hijo o una hija, mayores de edad, o por un siervo o una sierva, hebreos, o por su esposa después de divorciarla, aunque aún no le pagó su Ketubá (valor de la deuda contraída con ella en el contrato nupcial), les pertenece a ellos. (6) Encontraron documentos de endeudamiento, si aparecen en ellos garantías de bienes, no los pueden devolver (al acreedor), ya que el tribunal puede cobrarlos de ellos (de los deudores). Si no aparece en ellos garantías de bienes, los devolverá (al acreedor), ya que el tribunal no puede cobrarlos de ellos (de los compradores de la propiedad), palabras de Rabí Meir. Los demás Sabios objetan: tanto en uno como en otro caso no los devolverá, ya que el tribunal puede cobrarlos de ellos (de los compradores de la propiedad). (7) Si alguien encuentra documentos de divorcio de las mujeres, o de liberación de esclavos, o testamentos, o documentos de donaciones, o facturas, no deberá devolverlos a sus titulares, porque se podría argumentar: “Después de estar escritos seguro que se arrepintió y no llevó a efecto el documento”. (8) Si alguien encuentra certificados de peritación o de manutención, documentos de repudio o de rechazo, documentos de elección de los miembros del tribunal, o cualquier otro documento gestionado en un tribunal, deberá devolverlos. Si los encontró dentro de un odre o de un zurrón, mezclado en un manojo de documentos, o en un paquete de certificados, deberá devolverlos. ¿Cuántos certificados contiene un paquete? Tres unidos entre sí. Rabán Shimón Ben-Gamliel dice: Si se encuentran tres documentos de préstamo de un solo deudor de prestamistas distintos, los devolverá al deudor. Si se encuentran tres documentos de distintos deudores relacionados con un único prestamista, los devolverá al prestamista. Si encontró un documento de deuda entre sus documentos y desconoce su naturaleza, lo dejará aparte hasta que aparezca Eliyahu. Si entre ellos se encuentra documentos de liquidación de deudas, que lleve a cabo lo que queda especificado en dichos documentos.

    Rabí Ovadiá de Bartenura

    1. DOS AGARRAN UN TALIT. La Guemará interpreta la Mishná como si uno de ellos lo agarra por los flecos del borde de la prenda de un lado, y el otro lo agarra por los flecos de otro lado. Sin embargo, si lo que agarran es la prenda en sí, cada uno se llevará lo que tiene agarrado, y el resto lo dividirán en partes iguales y bajo juramento. UNO DICE: “ES TODO MÍO”. Yo la adquirí, a mí me lo vendió el vendedor y no a ti. En realidad, el vendedor vendió la prenda a uno de ellos, pero tomó dinero de ambos, de uno conscientemente y del otro involuntariamente, y no recuerda quien fue uno y otro. Si supiera con certeza a quien vendió, ya habría un testigo, lo que ocasionaría que el otro tuviera la obligación de juramentar en nombre de la Torá para desechar dicho testimonio. Como no lo sabe, los dos deben jurar. De acuerdo con la Ley, deberían repartirse el objeto en disputa sin necesidad de ningún juramento, pero los rabinos temieron que, entonces, cualquiera iba a agarrar algo del compañero y decir, sin ninguna temeridad, que le pertenece. Por otro lado, el exponente de la Mishná no tuvo más remedio que hablar de dos supuestos, uno de compra y el otro de hallazgo. Si sólo hubiese mencionado el supuesto del hallazgo, pensaríamos que los sabios lo obligan a jurar porque este pensaría: “Siendo que no le estoy quitando nada que le pertenezca a mi compañero, voy a agarrarlo para mí”, y con esto se justifica. En cambio, en el caso de compra mi compañero necesita el objeto, por eso fue a comprarlo, no sospechamos de él, y no lo hacemos jurar. Por otro lado, si sólo hubiera hablado del caso de compra, pensaríamos que los rabinos obligaron a jurar porque sospechan que diga: “Siendo que él pagó y yo pagué, como yo lo necesito más, que vaya él y lo adquiera en otro lugar”. Este argumento no cabe en el hallazgo, por eso no obligan a jurar. Esta es la razón para mencionar forzosamente los dos supuestos. UNO JURA QUE POSEE NO MENOS DE LA MITAD. No jura que todo es de él, como en el primer caso, porque no se lo van a otorgar. Por otro lado, si se le hace jurar que la mitad le pertenece, no se atendería a la verdad, siendo que él pretende quedarse con el total. Por tanto, él jura que no posee menos de la mitad, queriendo decir que, en verdad, todo es suyo pero siendo que no le creen, al menos hace constar que no menos de la mitad le pertenece. 2. DOS MONTABAN A LA VEZ UN MISMO ANIMAL. De aquí se desprende que montar un animal puede ocasionar su adquisición, aún en el caso que este no se mueva de su sitio. UNO LO MONTABA Y EL OTRO LO GUIABA. Se equipara el que lo monta al que lo dirige, sólo en el caso que el primero mueva los pies para que ande. Si sólo montaba, entonces lo adquiere el que lo dirige, no el que lo monta. Si, posteriormente, se ponen de acuerdo o aparecen testigos, aún después de haberse decretado que se lo repartan previo juramento, entonces, lo repartirán sin juramento. ALCÁNZAMELO. Pero si dijera: Tómalo para que yo me apodere de él, entonces, le pertenecería al jinete, y el que lo alzó no podría decir: A mí me pertenece. NO VALE LO QUE DIJO. La adquirió aquel que la tomó de manos del compañero. Mientras permanecía en manos de quien lo levantó no pertenecía a nadie. AQUEL QUE LO ASIÓ SE LO QUEDA. Hablamos de un objeto encontrado en la vía pública, donde el perímetro de cuatro codos de cada cual no otorga la propiedad, por esto, el que lo asió se lo queda. Sin embargo, en un callejón o en los confines de la vía pública, el perímetro de cuatro codos alrededor de una persona adquiere un objeto para él. Nadie puede quitárselo. Este decreto rabínico se hizo para evitar discusiones. UNA GACELA CON UNA PATA ROTA. Que no puede correr y se resguarda en el campo para que no la atrapen, entonces se considera como algo encontrado. LE PERTENECE. El hecho de que el campo le pertenece le otorga derecho sobre lo que en él se encuentre. Esto es así siempre que se encuentre dentro en un lugar tal, que si sale corriendo a atraparlo lo podría conseguir antes de salir de los confines de este. 5. MENORES DE EDAD. Aquel que es mantenido por el padre, aun siendo mayor de edad, es denominado “menor de edad”. Lo que encuentra se lo da al padre para no despertar enemistades. Si el hallazgo es de la hija, tanto menor como doncella, le pertenece al padre. La Torá indicó que todo lo que ella posee hasta su mayoría de edad, le pertenece al padre. UN SIERVO O UNA SIERVA, QUENAANITAS. Ya que ellos mismos le pertenecen. Pues así está escrito: “Los heredaréis” (Levítico 25:46). POR SU ESPOSA. Los rabinos decretaron esta disposición para evitar pleitos. 6. GARANTÍAS DE BIENES. Hipoteca de bienes inmuebles para garantizar el cobro. NO LOS PUEDEN DEVOLVER. Porque existen sospechas por el pago y por la compra. Podría haber ocurrido que el documento estaba pagado y que se le cayó al deudor. Incluso que el deudor reconozca que no lo pagó, no se puede devolver, no sea que entre ambos, el acreedor y el deudor, hayan tramado engañar a los posibles compradores. El deudor podría haber vendido propiedades hipotecadas a un tercero. Después, se ponen de acuerdo, acreedor y deudor, para apoderarse de dichas propiedades vendidas, con la excusa de estar hipotecadas y después se reparten las ganancias. TANTO EN UNO COMO EN OTRO CASO. Porque, aún un documento que no esté garantizado con bienes también es cobrable. La responsabilidad está tácitamente implícita en cualquier contrato aunque no esté especificada, y si no aparece, lo achacamos a que se le olvidó al escriba reflejarlo. La Ley se aplica según la opinión de los Sabios. Precisamente, en aquellos documentos donde no se especifica la responsabilidad decimos que hubo un error de redacción, pero si se aclara de forma concisa que no exige ninguna responsabilidad, también los Sabios concuerdan que debe devolver el documento encontrado, pues aquí no se teme nada más. TESTAMENTOS. Documentos de herencia redactados por personas moribundas. FACTURAS. Documentos que el prestamista extiende al prestatario. 8. DOCUMENTOS DE PERITACIÓN. Evaluación que realiza el tribunal sobre los bienes del prestatario. MANUTENCIÓN. Documento en el que el esposo se compromete a alimentar a la hija de su esposa. Según otra explicación, se refiere a los bienes inmuebles que debe vender el esposo para mantener a su esposa e hijas. RECHAZO. Se escribe en un documento: En nuestra presencia rechazó Mengana a Fulano, su esposo. También, el rechazo que manifiesta una niña casada por mandato de su madre y hermanos y que no requiere documento de divorcio. ELECCIÓN DEL TRIBUNAL. La elección de magistrados que realiza cada uno de los litigantes en un juicio. ODRE. Bolsa pequeña de cuero. ZURRÓN. Bolsa de cuero utilizada por los ancianos para guardar en ella sus utensilios y así no tener que buscarlos. MANOJO DE DOCUMENTOS. Tres o más documentos atados juntos. PAQUETE. Certificados amontonados unos sobre otros. DEBERÁ DEVOLVERLOS. Siendo que están identificados por la bolsa que los contiene. TRES DOCUMENTOS DE PRÉSTAMO. Si los tres documentos de préstamo de un único prestatario pertenecen a tres diferentes prestamistas, deberá devolverlos al prestatario, pues seguro se le cayó. No cabe duda de que si pertenecieran a los prestamistas no hubiesen caído los tres en un mismo lugar. Además, los documentos deben estar ratificados por el Tribunal, en caso contrario, podríamos pensar que los prestamistas se dirigieron al tribunal para dicho trámite y se le perdieron al escriba del Tribunal. Descartamos que estuvieran ratificados, porque en ese caso no permanecerían en manos del escriba sino con sus respectivos dueños. Si fueran de tres prestatarios diferentes que tomaron de un solo prestamista, el que los encontró deberá devolverlos al prestamista, seguro que a él se le cayeron. Si los tres documentos muestran la escritura de un mismo escriba, pensaríamos que los estaba redactando dicho escriba y que nunca se realizaron, por esto no los devolverá. SI ENCONTRÓ UN DOCUMENTO... En su poder. No sabe si fue el prestamista o el prestatario quien se lo dio, o si está parcialmente pagado. LO DEJARÁ. En sus manos, no lo devolverá ni al uno ni al otro. DOCUMENTOS DE LIQUIDACIÓN. Si alguien encuentra entre sus documentos una factura sobre documentos propios... QUE LLEVE A CABO... Se considera entonces, potencialmente, pagada la factura. A pesar de que este documento debería estar en manos del prestatario y no en manos del prestamista, pensamos que el prestatario confiaba en el prestamista. Seguro le dijo: Déjalo aquí hasta mañana, entonces, devuélvemelo. Y se olvidó. Este caso se refiere a que el prestamista encontró dicho documento entre otros rotos, aunque el mismo no lo estaba.

    Guemará

    Tratado Babá-Metsiá 29b

    Estudiaron nuestros Sabios: «Quien pide prestado un Séfer Torá de un amigo no debe prestarlo a otra persona. Lo puede abrir para leerlo, con la condición de que sea un tema por primera vez estudiado. Tampoco podrá leerlo nadie junto a él. Un Séfer entregado en custodia lo deben hacer girar, al menos, una vez cada doce meses. Al abrirlo puede leer en él, pero abrirlo expresamente para sí mismo con la intención de leerlo está prohibido. Somjós dice: Un Séfer nuevo puede abrirlo cada treinta días, uno viejo, cada doce meses. Rabí Eliézer Ben Yaacob dijo: Tanto uno como otro, cada doce meses». Dijo cierto rabino anteriormente: «Quien pide prestado un Séfer Torá de un amigo no debe prestarlo a otra persona». ¿Por qué dijo esto sólo con relación al Séfer Torá? ¡En todos los casos está prohibido! Así dijo Ribí Shimón Ben Lajish: Aquí, nos enseñó Ribí: “Una cosa prestada no se puede prestar, y una cosa alquilada no se puede alquilar”. Fue necesario especificar el Séfer, porque podrías argumentar: “Todo el mundo disfruta cuando alguien utiliza algo de su propiedad para cumplir un precepto”. De aquí aprendemos que no es así. Se dijo: «Lo puede abrir para leerlo». ¡Evidente! Si no,¿para qué lo pide prestado? Estas palabras se refieren a lo del final, o sea, «Lo puede abrir para leerlo, pero no para estudiar algo por primera vez». Asimismo, se dijo: «Un Séfer entregado en custodia lo deben hacer girar, al menos, una vez cada doce meses. Al abrirlo puede leer en él...». ¿Con qué finalidad? Además, se dijo: «Abrirlo expresamente para sí mismo con la intención de leerlo está prohibido». Entonces, ¿cómo dijo antes: «Lo puede abrir para leerlo»? Así quiso decir: “Si mientras hace girar el Séfer lo abre y lo lee, está permitido, pero si lo abre expresamente para él, está prohibido”. «Somjós dice: Un Séfer nuevo puede abrirlo cada treinta días, uno viejo, cada doce meses. Rabí Eliézer Ben Yaacob dijo: Tanto uno como otro, cada doce meses». ¿Acaso podemos afirmar que Rabí Eliézer Ben Yaacob opina como el primer exponente de la Mishná? No, sino que el texto exacto sería: “Rabí Eliézer Ben Yaacob dijo: Tanto uno como otro, cada treinta días”. «Con la condición de que sea un tema por primera vez estudiado. Tampoco podrá leerlo nadie junto a él». Sobre esto se alega lo siguiente: Está dicho: “No se puede leer una Perashá y repetirla. No se puede leer una Parashá y después traducirla. No deberá abrir más de tres columnas. No deben leer en él tres personas a la vez”. Según esto, ¡dos sí pueden! Dijo Abayé: No es contradictorio, depende si leen un solo asunto o dos.

    Vayigash 208b

    Ven y observa. Dijo Eliyahu: “¿Acaso también a la viuda, en cuya casa yo me hospedo, vas a dañar matando a su hijo?” (Reyes I 17:20). El Santo, Bendito Él, ya le había informado a Eliyahu que una mujer viuda le iba a procurar alimento (ibíd.). Y, es sabido, que quien alimenta o sustenta a quien lo necesita, y más aún en tiempos de hambre, se conecta al Árbol de la Vida, procurándole vida a él y a sus hijos. Así está determinado. Ahora dijo Eliyahu, que quien sustenta a un alma en este mundo obtiene como recompensa la vida, y merece conectarse con el Árbol de la Vida. Sin embargo, ahora domina el Árbol de la Muerte, el Lado del Mal, sobre aquella viuda que mandaste para que me sustentara. Por esto dijo: “a la que vas a dañar”. ¿Acaso podrías argumentar que el mal que daña al hombre no procede del Santo, Bendito Él? Ven y observa. En aquella ocasión en la cual el hombre se dirige hacia la derecha, la Protección Divina siempre lo acompaña, y, por tanto, el Lado Oculto nunca puede gobernar sobre él. Como consecuencia de esto, el Mal queda humillado y no puede dominarlo. Por el contrario, si la Protección Divina se aleja de la persona por haberse apegado al Mal, entonces, el Mal, siendo que no ve ninguna Protección, lo domina y procura aniquilarlo, siendo así, se le otorga el permiso para que se lleve su alma. Moshé dijo: “¿Por qué dañaste a este pueblo?” (Éxodo 5:22), debido a que se le dio permiso al Lado del Mal para gobernar sobre Israel, para subyugarlo. Según otra explicación, “¿Por qué dañaste a este pueblo?”, lo dijo, porque vio a muchos de Israel muertos, que fueron entregados en las manos del Lado del Mal. Ven y observa. Cuando el Bien se despierta, esto es, la Diestra, se da lugar a todas las alegrías y todas las bendiciones, pero todo en voz baja. Así ya aclaramos, que se dice “Baruj Shem Kebod Maljutó Leolam Vaed”, en voz baja. El secreto de esto se debe a que se realiza la unión más idónea.

    Halajá

    Rambam, Leyes sobre el Arrepentimiento, Capítulo 2

    ¿Cómo? Un hombre cometió un pecado con cierta mujer. Al cabo de un tiempo, se encuentra con ella, sintiendo el mismo amor, con las mismas fuerzas, en el mismo lugar, sin embargo, se aparta de ella y no peca. Esto se denomina un “completo arrepentimiento”. Sobre esto dijo el Rey Salomón: “Acuérdate de tu Creador en tu juventud” (Eclesiastés 12:1). (2) Si su arrepentimiento llega en la vejez, en aquel momento donde no podía repetir lo que hizo con anterioridad, aunque no se considere un arrepentimiento de categoría, le sirve y se le denomina “arrepentido”. Aunque una persona transgreda los preceptos durante toda su vida, si se arrepiente el mismo día de su muerte y muere arrepentido, se le perdonan todos los pecados, tal como versa: “Hasta que no se oscurezca el sol, y la luz, la luna y las estrellas, y retornen las nubes después de la lluvia” (ibíd. 12:2). Esto se refiere al día de la muerte. De aquí se deduce, que si se acordó de su Creador y se arrepintió antes de morir, queda perdonado. (3) ¿En qué consiste el arrepentimiento? El pecador abandonará sus pecados, los eliminará de su mente, y determinará en su corazón no volver a hacerlos más. Así está dicho: “Abandonará el malvado su camino” (Isaías 55:7). Asimismo, se arrepentirá de su pasado, tal como versa: “Cuando volví me arrepentí” (Jeremías 31:18). Pondrá como testigo al Conocedor de los Secretos que no volverá a cometer tal trasgresión, pues así se dijo: “Y no recurriremos más a los dioses, obra de nuestras manos”. (Oseas 14:4). Además, deberá confesarse con sus labios y comentar aquellas cosas que determinó en su corazón no hacer más.

    Moral

    Séfer Hajaredím, 71a

    Con respecto a la arrogancia está escrito: “Detestable para el Eterno es todo engreído de corazón” (Proverbios 16:5). Nuestros Maestros, de bendita memoria, dijeron: “El arrogante es rechazado”, o sea, rechazado por El Eterno y rechazado por los Emoraim. Sobre el enojo está dicho: “No te harás dioses fundidos” (Éxodo 34:17), porque el que se enoja es como si adorara ídolos, y todo tipo de infiernos lo dominan. Está prohibido relacionarse con personas muy irascibles porque le ocasiona la impureza de sus doscientos cuarenta y ocho órganos, su alma se esfuma y, en su lugar, se coloca la impureza, como así aparece en el Zóhar al comienzo de la Parashá de Tetzavé: “No odies a tu hermano con tu corazón” (Levítico 19:17), que es la raíz de todos los males. Fíjate lo que ocasionó el odio a Yosef por parte de sus hermanos. Por culpa del odio gratuito padecemos el exilio más largo de todos. Incluso aquel que se decretó como consecuencia de la idolatría, las relaciones pecaminosas y el asesinato, en la época del Primer Templo. Así como “amarás a tu prójimo como a ti mismo” (Levítico 19:18), incluye el cumplimiento de toda la Torá, el odio produce el efecto contrario, esto es, transgrede toda la Torá. “El corazón que trama maldades” (Proverbios 6:18) es aborrecido por el Eterno, tal como está escrito: “Hay seis cosas que desprecia el Eterno” (ibíd. 6:16). Nuestros Maestros, de bendita memoria, dijeron: Los pensamientos pecaminosos son peores que el pecado. Escribió sobre esto Ribí Abraham Ibn-Ezra que “impurifican el alma que es pura, y que es la antesala del pecado, así como arar lo es ante el plantar”. Aunque la persona no tenga la intención de cometer el pecado, el sólo hecho de pensarlo y sentirlo en su corazón, de imaginarlo, perjudica y distancia al hombre de su Creador. Al respecto advirtió la Torá: “No os dejéis llevar tras vuestros corazones y vuestros ojos” (Números 15:39). Quien imagina cosas pecaminosas es como quien entra a un lugar sucio y sus ropas se impregnan de suciedad. ¿No sentirá vergüenza de presentarse así ante el Rey? Sobre esto se dijo: “Lavaos, purificaos” (Isaías 1:16). (Que el Eterno nos purifique y nos otorgue el privilegio de cumplir Su Voluntad).
