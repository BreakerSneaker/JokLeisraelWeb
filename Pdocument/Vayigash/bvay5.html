    ה וְעַתָּ֣ה | אַל־תֵּעָ֣צְב֗וּ וְאַל־יִ֨חַר֙ בְּעֵ֣ינֵיכֶ֔ם כִּֽי־מְכַרְתֶּ֥ם אֹתִ֖י הֵ֑נָּה כִּ֣י לְמִֽחְיָ֔ה שְׁלָחַ֥נִי אֱלֹהִ֖ים לִפְנֵיכֶֽם: וּכְעַן לָא תִתְּנְסוּן וְלָא יִתְקָף בְּעֵינֵיכוֹן אֲרֵי זַבִּנְתּוּן יָתִי הָכָא אֲרֵי לְקַיָּמָא שַׁלְּחַנִי יְיָ קֳדָמֵיכוֹן: ו כִּי־זֶ֛ה שְׁנָתַ֥יִם הָֽרָעָ֖ב בְּקֶ֣רֶב הָאָ֑רֶץ וְעוֹד֙ חָמֵ֣שׁ שָׁנִ֔ים אֲשֶׁ֥ר אֵין־חָרִ֖ישׁ וְקָצִֽיר: אֲרֵי דְנַן תַּרְתֵּין שְׁנִין כַּפְנָא בְּגוֹ אַרְעָא וְעוֹד חָמֵשׁ שְׁנִין דִּי לֵית זְרוֹעָא וַחֲצָדָא: ז וַיִּשְׁלָחֵ֤נִי אֱלֹהִים֙ לִפְנֵיכֶ֔ם לָשׂ֥וּם לָכֶ֛ם שְׁאֵרִ֖ית בָּאָ֑רֶץ וּלְהַֽחֲי֣וֹת לָכֶ֔ם לִפְלֵיטָ֖ה גְּדֹלָֽה: וְשַׁלְּחַנִי יְיָ קֳדָמֵיכוֹן לְשַׁוָּאָה לְכוֹן שְׁאָרָא בְּאַרְעָא וּלְקַיָּמָא לְכוֹן לְשֵׁיזָבָא רַבָּא: שלישי ח וְעַתָּ֗ה לֹֽא־אַתֶּ֞ם שְׁלַחְתֶּ֤ם אֹתִי֙ הֵ֔נָּה כִּ֖י הָֽאֱלֹהִ֑ים וַיְשִׂימֵ֨נִי לְאָ֜ב לְפַרְעֹ֗ה וּלְאָדוֹן֙ לְכָל־בֵּית֔וֹ וּמשֵׁ֖ל בְּכָל־אֶ֥רֶץ מִצְרָֽיִם: וּכְעַן לָא אַתּוּן שְׁלַחְתּוּן יָתִי הָכָא אֶלָּהֵן מִן קֳדָם יְיָ וְשַׁוְּיַנִי לְאַבָּא לְפַרְעֹה וּלְרִבּוֹן לְכָל אֱנַשׁ בֵּיתֵהּ וְשַׁלִּיט בְּכָל אַרְעָא דְמִצְרָיִם: ט מַֽהֲרוּ֘ וַֽעֲל֣וּ אֶל־אָבִי֒ וַֽאֲמַרְתֶּ֣ם אֵלָ֗יו כֹּ֤ה אָמַר֙ בִּנְךָ֣ יוֹסֵ֔ף שָׂמַ֧נִי אֱלֹהִ֛ים לְאָד֖וֹן לְכָל־מִצְרָ֑יִם רְדָ֥ה אֵלַ֖י אַל־תַּֽעֲמֹֽד: אוֹחוּ וּסְקוֹ לְוַת אַבָּא וְתֵימְרוּן לֵהּ כִּדְנַן אֲמַר בְּרָךְ יוֹסֵף שַׁוְּיַנִי יְיָ לְרִבּוֹן לְכָל מִצְרָיִם חוּת לְוָתִי לָא תִתְעַכָּב:

    רש"י

    (ה) למחיה. להיות לכם למחיה: (ו) כי זה שנתיים הרעב. עברו משני הרעב: (ח) לאב. לחבר ולפטרון: (ט) ועלו אל אבי. ארץ ישראל גבוה מכל הארצות:

    נביאים

    יחזקאל פרק ל"ח

    ח מִיָּמִ֣ים רַבִּים֘ תִּפָּקֵד֒ בְּאַחֲרִ֨ית הַשָּׁנִ֜ים תָּב֣וֹא | אֶל־אֶ֣רֶץ | מְשׁוֹבֶ֣בֶת מֵחֶ֗רֶב מְקֻבֶּ֙צֶת֙ מֵעַמִּ֣ים רַבִּ֔ים֚ עַל הָרֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁר־הָי֥וּ לְחָרְבָּ֖ה תָּמִ֑יד וְהִיא֙ מֵעַמִּ֣ים הוּצָ֔אָה וְיָשְׁב֥וּ לָבֶ֖טַח כֻּלָּֽם: מִיּוֹמִין סַגִּיאִין תְּמַנֵּי מַשִׁרְיָתָךְ בְּסוֹף שְׁנַיָּא תְּהַךְ עַל אַרְעָא דְּתָבוּ עֲלָהּ מְקַטְלֵי חַרְבָּא דְּאִתְכַּנְּשׁוּ מֵעַמְמַיָּא סַגִּיאִין עַל טוּרֵי אַרְעָא דְיִשְׂרָאֵל דַּהֲווֹ לְחוּרְבָּא תְּדִירָא וְאִינּוּן מִבֵּינֵי עַמְמַיָּא אִתְגְּלִיאוּ וְיָתִיבוּ לְרוֹחֲצָן כֻּלְּהוֹן: ט וְעָלִ֙יתָ֙ כַּשּׁוֹאָ֣ה תָב֔וֹא כֶּעָנָ֛ן לְכַסּ֥וֹת הָאָ֖רֶץ תִּֽהְיֶ֑ה אַתָּה֙ וְכָל־אֲגַפֶּ֔יךָ וְעַמִּ֥ים רַבִּ֖ים אוֹתָֽךְ: וְתִסַק בְּאִתְרְגוּשָׁתָא תֵּיתֵי כַּעֲנָנָא דְּסַלֵיק וְחָפֵי יַת אַרְעָא תְּהֵי אַתְּ וְכָל מַשִׁרְיָתָךְ וַעֲמָמִין סַגִּיאִין דִּי עִמָּךְ: י כֹּ֥ה אָמַ֖ר אֲדֹנָ֣י יֱהֹוִ֑ה וְהָיָ֣ה | בַּיּ֣וֹם הַה֗וּא יַעֲל֤וּ דְבָרִים֙ עַל־לְבָבֶ֔ךָ וְחָשַׁבְתָּ֖ מַחֲשֶׁ֥בֶת רָעָֽה: כִּדְנַן אֲמַר יְיָ אֱלֹהִים וִיהֵי בְּעִדָנָא הַהִיא יִסְּקוּן הִרְהוּרִין עַל לִבָּךְ וְתַחֲשִׁיב מַחְשְׁבָן בִּישָׁן: יא וְאָמַרְתָּ֗ אֶֽעֱלֶה֙ עַל־אֶ֣רֶץ פְּרָז֔וֹת אָבוֹא֙ הַשֹּׁ֣קְטִ֔ים ישְׁבֵ֖י לָבֶ֑טַח כֻּלָּ֗ם יֽשְׁבִים֙ בְּאֵ֣ין חוֹמָ֔ה וּבְרִ֥יחַ וּדְלָתַ֖יִם אֵ֥ין לָהֶֽם: וְתֵימַר אֶסַּק עַל אַרְעָא דְּיָתְבִין שְׁלֵיוָא בְּקִרְוֵי פִּצְחַיָּא אֵיהַךְ עַל עַמָּא דְּשָׁקְטִין וּשְׁרָן לְרוֹחֲצָן כֻּלְּהוֹן יָתְבִין בְּלָא שׁוּרָא וְעָבְרִין וְדָשִּׁין לֵית לְהוֹן: יב לִשְׁלֹ֥ל שָׁלָ֖ל וְלָבֹ֣ז בַּ֑ז לְהָשִׁ֨יב יָדְךָ֜ עַל־חֳרָב֣וֹת נוֹשָׁבֹ֗ת וְאֶל־עַם֙ מְאֻסָּ֣ף מִגּוֹיִ֔ם עֹשֶׂה֙ מִקְנֶ֣ה וְקִנְיָ֔ן ישְׁבֵ֖י עַל־טַבּ֥וּר הָאָֽרֶץ: לְמֵיבַז בֵּיזָא וּלְמַעֲדֵּי עֲדָאָה לְכַנְשֵׁי מַשִׁרְיָתָךְ עַל חָרְבָתָא דְּאִתְיַתְבָא וְעַל עַמָּא דְּאִתְכַּנְשׁוּן מִבֵּינֵי עַמְמַיָּא דְּאַצְלָחוּ בְּנִכְסִין וְקִנְיָנִין יָתְבִין עַל תּוּקְפָא דְאַרְעָא:

    רש"י

    (ח) מִיָּמִים רַבִּים שֶׁחָטָאתָ לִי: תִּפָּקֵד עַתָּה: מְשׁוֹבֶבֶת מֵחֶרֶב אֶל אֶרֶץ יִשְׂרָאֵל שֶׁשָּׁבוּ יוֹשְׁבֶיהָ מִן הַגּוֹלָה: (י) מַחֲשֶׁבֶת רָעָה לְהִלָּחֵם כְּנֶגְדִּי: (יא) אֶרֶץ פְּרָזוֹת מֵאֵין חוֹמָה: אָבוֹא הַשֹּׁקְטִים וְאֶרְאֶה מַה הוּא מִבְטָחָם: (יב) מְאֻסָּף מִגּוֹיִם וְעוֹשִׂים עַתָּה מִקְנֶה וְקִנְיָן: עַל טַבּוּר הָאָרֶץ עַל גֹּבַהּ וְחֹזֶק הָאָרֶץ:

    כתובים

    משלי י

    כו כַּחֹ֤מֶץ | לַשִּׁנַּ֗יִם וְכֶעָשָׁ֥ן לָעֵינָ֑יִם כֵּ֥ן ֜הֶעָצֵ֗ל לְשֹׁלְחָֽיו: הֵיךְ חוּמְצָא לְשִׁנֵּי וְכִתְנָנָא לְעַינֵי הֵיכְנָא שְׁלִיחָא עַטְלָא לִמְשַׁלְּחוֹי: כז יִרְאַ֣ת ֖יְהֹוָה תּוֹסִ֣יף יָמִ֑ים וּשְׁנ֖וֹת רְשָׁעִ֣ים תִּקְצֹֽרְנָה: דְּחַלְתֵּיהּ דֶּאֱלָהָא תוֹסֵף יוֹמָתָא וּשְׁנֵי דְרַשִּׁיעֵי נִזְעֲרָן: כח תּוֹחֶ֣לֶת צַדִּיקִ֣ים שִׂמְחָ֑ה וְתִקְוַ֖ת רְשָׁעִ֣ים תֹּאבֵֽד: סִבְרְהוֹן דְּצַדִיקֵי חֶדְוְתָא וְסִבְרָא דְרַשִּׁיעֵי נֵאבַד: כט מָעֹ֣ז ֖לַתֹּם דֶּ֣רֶךְ יְהֹוָ֑ה ֜וּמְחִתָּ֗ה לְפֹ֣עֲלֵי אָֽוֶן: מַעְשְׁנָא לִתְמִימֵי אוֹרְחֵיהּ דֶּאֱלָהָא וְשֵׁצְיָא נֶהֱוֵי לְאִלֵין דְּעָבְדִין עַוְתָא: ל צַדִּ֣יק לְעוֹלָ֣ם בַּל־יִמּ֑וֹט ֜וּרְשָׁעִ֗ים לֹ֣א יִשְׁכְּנוּ־אָֽרֶץ: צַדִיק לַעֲלָם לָא נְזוּעַ וְרַשִּׁיעֵי לָא יִתְוַעֲדוּן בְּאַרְעָא:

    רש"י

    (כח) תּוֹחֶלֶת צַדִּיקִים שִׂמְחָה. סוֹפָה מִתְקַיֶּמֶת וּשְׂמֵחִים: וְתִקְוַת רְשָׁעִים תֹּאבֵד. כִּי לֹא תָבֹא: (כט) מָעֹז לַתֹּם דֶּרֶךְ ה' וּמְחִתָּה לְפֹעֲלֵי אָוֶן. שֶׁאֵינָן הוֹלְכִין בָּהּ, וְהִיא נִפְרַעַת מֵהֵם: (ל) צַדִּיק לְעוֹלָם בַּל יִמּוֹט. כְּשֶׁהוּא מַט, אֵין מוֹטַתוֹ מוֹטַת עוֹלָם, כִּי יִפֹּל וְקָם:

    משנה

    זבחים פרק יא

    גמרא

    זבחים דף קט"ו ע"ב

    תַּנְיָא (שמות יט) וְגַם הַכֹּהֲנִים הַנִּגָּשִׁים אֶל ה' יִתְקַדָּשׁוּ. רַבִּי יְהוֹשֻׁעַ בֶּן קָרְחָה אוֹמֵר זוּ פְּרִישׁוּת בְּכוֹרוֹת. רַבִּי אוֹמֵר זוּ פְּרִישׁוּת נָדָב וַאֲבִיהוּא. בִּשְׁלָמָא לְמַאן דְּאָמַר זוּ פְּרִישׁוּת נָדָב וַאֲבִיהוּא הַיְינוּ דִּכְתִיב (ויקרא י') הוּא אֲשֶׁר דִּבֶּר ה' לֵאמֹר בִּקְרוֹבַי אֶקָּדֵשׁ. אֶלָּא לְמַאן דְּאָמַר זוּ פְּרִישׁוּת בְּכוֹרוֹת. הֵיכָא רְמִיזָא דִּכְתִיב (שמות כג) וְנוֹדַעְתִּי שָׁמָּה לִבְנֵי יִשְׂרָאֵל וְנִקְדַּשׁ בִּכְבוֹדִי אַל תִּקְרֵי בִּכְבוֹדִי אֶלָּא בִּמְכוּבָּדַי. דָּבָר זֶה רָמַז הַקָּדוֹשׁ בָּרוּךְ הוּא לְמשֶׁה וְלֹא יָדַע עַד שֶׁמֵּתוּ בְּנֵי אַהֲרֹן. כִּיוָן שֶׁמֵּתוּ בְּנֵי אַהֲרֹן אָמַר לוֹ. אַהֲרֹן אָחִי לֹא מֵתוּ בָנֶיךָ אֶלָּא לְהַקְדִּישׁ שְׁמוֹ שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא כִּיוָן שֶׁיָּדַע אַהֲרֹן שֶׁבָּנָיו יְדוּעֵי מָקוֹם הֵן שָׁתַק וְקִבֵּל שָׂכָר שֶׁנֶּאֱמַר (ויקרא י) וַיִּדּוֹם אַהֲרֹן וְכֵן בְּדָוִד הוּא אוֹמֵר (תהלים לז) דּוֹם לַה' וְהִתְחוֹלֵל לוֹ אַף עַל פִּי שֶׁמַּפִּיל לָךְ חֲלָלִים חֲלָלִים אַתָּה שְׁתוֹק. וְכֵן בִּשְׁלֹמֹה הוּא אוֹמֵר (קהלת ג') עֵת לַחֲשׁוֹת וְעֵת לְדַבֵּר פְּעָמִים שֶׁשּׁוֹתֵק וּמְקַבֵּל שָׂכָר עַל הַשְּׁתִיקָה וּפְעָמִים שֶׁמְּדַבֵּר וּמְקַבֵּל שָׂכָר עַל הַדִּבּוּר וְהַיְינוּ דְאָמַר רַבִּי חִיָּיא בַּר אַבָּא אָמַר רַבִּי יוֹחָנָן מָאי דִּכְתִיב (תהלים סח) נוֹרָא אֱלֹהִים מִמִּקְדָּשֶׁיךָ אַל תִּקְרֵי מִמִּקְדָשֶׁיךָ אֶלָּא מִמְּקוּדָשֶׁיךָ בְּשָׁעָה שֶׁעוֹשֶׂה הַקָּדוֹשׁ בָּרוּךְ הוּא דִּין בִּקְדוֹשָׁיו מִתְיָרֵא וּמִתְעַלֶּה וּמִתְהַלָּל:

    תרגום הגמרא

    למדנו בברייתא: "וגם הכהנים הנגשים אל ה' יתקדשו" (שמות י"ט). רבי יהושע בן קרחה אומר: זו פרישות בכורות, כלומר זו אזהרה לבכורות שאף הם יפרישו עצמם ואל יעלו להר סיני ואף על פי שהם רגילים לגשת אל ה' ולהעלות קרבנות, לפי שלפני הקמת המשכן נעשתה עבודה ע"י הבכורות. רבי אומר: זו פרישות נדב ואביהוא, כלומר לא נקראו הבכורות 'כהנים', אלא כוונת הכתוב לנדב ואביהוא בני אהרן שנעשו כהנים בסיני והם הכהנים הנגשים אל ה' להקריב קרבנות. לדעת רבי שהכתוב מדבר בנדב ואביהוא, מובן מה שנאמר לאחר מיתתם "הוא אשר דבר ה' לאמר בקרובי אקדש", כלומר ה' רמז על כך בסיני כמו שכתוב: "וגם הכהנים הנגשים אל ה' יתקדשו פן יפרוץ בהם ה'" ומכיון שלא נזהרו לפרוש מן ההיכל בהקמת המשכן- פרץ בהם ה' ומתו. אבל לדעת רבי יהושע בן קרחה שהכתוב מדבר בפרישות בכורות מהר סיני, היכן נרמזה למשה קודם לכן מיתת נדב ואביהוא כפי שנאמר בעת מות שני בני אהרן: הוא אשר דבר ה' לאמר בקרובי אקדש? שכתוב (שמות כ"ט) "ונועדתי שמה לבני ישראל ונקדש בכבודי", אל תקרא בכבודי אלא במכובדי, "ונועדתי שמה" הכוונה היא למשכן ורמז לו שביום שישרה שכינתו על הבית יתקדש במכובדיו. דבר זה אמר הקב"ה למשה ולא ידעו, כלומר משה לא הבין את הדברים, עד שמתו בני אהרן, כיון שמתו בני אהרן אמר לו משה לאהרן: אהרן אחי, לא מתו בניך אלא להקדיש שמו של הקב"ה, כיון שידע אהרן שבניו ידועי מקום הם, כלומר שנבחרו לקדש בהם את ה' – שתק וקבל שכר שנאמר (ויקרא י') "וידום אהרן", וכן בדוד הוא אומר (תהלים ל"ז) "דום לה' והתחולל לו" כלומר אף על פי שה' מפיל לך חללים חללים, אתה שתוך ואל תבוא בטרוניא כךפי מעלה. וכן בשלמה הוא אומר (קהלת ג') "עת לחשות ועת לדבר" פעמים ששותק ומקבל שכר על השתיקה ופעמים שמדבר ומקבל שכר על הדיבור. וזהו מה שאמר רבי חייא בר אבא משמו של רבי יוחנן: מהו שכתוב (תהלים ס"ח) "נורא אלקים ממקדשך" – אל תקרא ממקדשךך אלא ממוקדשיך, בשעה שעושה הקב"ה דין בקדושיו (לפי ש"אלקים" זו מדת הדין) הוא מתיירא (כלומר מוטלת מוראו על הבריות) ומתעלה ומתהלל (משתבח).

    רש"י

    זו פרישות בכורות הזהיר את הבכורות להפריש מן ההר: זו פרישות נדב ואביהוא כלומר לא נקראו בכורות כהנים אלא מי הן הכהנים הנגשים שהוזהרו כאן לפרוש מן ההר זה נדב ואביהוא שנעשו בסיני כהנים והן הכהנים הנגשים אל ה' להקריב קרבנות: היינו דכתיב כשמתו נדב ואביהוא הוא אשר דבר וגו' כדקאמר בסיני פן יפרוץ בהם ה' וכיון שלא הוזהרו לפרוש מן ההיכל בהקמת המשכן פרץ בהם: אלא למאן דאמר זו פרישות בכורות היכא רמיזא. הכי גרסינן ולא גרסינן פרישת נדב ואביהוא: היכא רמיזא מיתת נדב ואביהוא היכא רמזה למשה קודם לכן דקאמר הוא אשר דבר: ונקדש בכבודי נאמרה לו למשה ביום שנצטווה על מלאכת המשכן ורמז לו שביום שישרה שכינתו על הבית יתקדש בכבודו ולא נרמז להבין מה הוא עד שמתו. אלא להתקדש כבוד של מקום עליהם להודיע שהוא נורא על יראיו ומיודעיו וזהו אשר דבר אלי ביום שנצטויתי על המשכן ונקדש בכבודי באותם הנכבדים וקרובים לי אקדש: שנאמר וידום אהרן הרי ששתק וקבל השכר מפ' בב"ר שמתוך כך זכה ונתייחד אליו הדיבור שנאמר אחרי אותו ענין פ' שתויי יין ולא הוזכר באותו דיבור משה אלא אהרן: נורא אלהי' ממקדשיך אלהים זו מדת הדין אימתי הוא נורא על ידי שהוא עושה דין במקודשיו:

    Mishná

    Tratado Zebajim, Capítulo 11

    (1) La sangre de un Sacrificio Jatat (Expiatorio) que salpica sobre una prenda, debe ser lavada, a pesar de que el texto solo se refiere a aquellos que son ingeribles, tal como versa: “en un lugar sagrado será ingerido” (Levítico 6:19). No obstante, tanto la ofrenda ingerible como la que se queda adentro, si salpican, debe ser lavado, pues así está dicho: “La Ley del Jatat” (ibíd. 6:18), o sea, una sola Ley para todos los Jatat. (2) La sangre de un Jatat inválido no requiere ser lavada, tanto si a la hora de salpicar la sangre el sacrificio era Kasher, como si ya no lo era. ¿Cuándo decimos que era Kasher el sacrificio? Cuando pasó la noche y se invalidó, cuando se impurificó el Jatat antes de salpicar y cuando salió la carne o la sangre del sacrificio, de la Azará antes de la salpicadura. ¿Cuándo decimos que no era Kasher en el momento de salpicar? Cuando fue sacrificado fuera de tiempo, o fuera de su lugar, o fue recibida dicha sangre por personas no aptas. (3) La sangre del animal que salió salpicada desde su cuello y manchó una prenda no se exige que sea lavada. Si salpicó desde un vértice del altar o desde la base, tampoco se exige que sea lavada. Si se derramó sobre el suelo y la recogió, y manchó una prenda, tampoco requiere ser lavada, pues sólo requiere ser lavada aquella sangre que fue recogida en el recipiente apropiado y que está destinada a ser salpicada. Si salpicó sobre la piel del animal antes de ser despellejado, no requiere lavado. Después de despellejado, sí lo requiere, según la opinión de Rabí Yehudá. Rabí Eliézer dice: Aún después de despellejado no requiere ser lavada. Solo requiere ser lavado el lugar exacto de la prenda manchada, aquello que puede contagiarse de la impureza y aquello que se puede lavar. (4) Tanto una prenda de tela, como un saco, como la piel, deben ser lavadas en el lugar sagrado. Asimismo, la rotura del recipiente de arcilla donde se cocina la carne debe realizarse en un lugar sagrado. También el lavado y enjuague del recipiente de cobre se hará en el lugar sagrado. Este tema se refiere solo al Jatat, ni siquiera a los Kodshé-Kodashim. (5) La prenda que salga de los límites de las cortinas debe ser de nuevo introducida y lavada en el lugar sagrado. Si se impurificó fuera de las cortinas, la rasgará, la introducirá y la lavará en un lugar sagrado. El recipiente de arcilla que salga de las cortinas deberá ser introducido y, posteriormente, quebrado en un lugar sagrado. Si se impurificó fuera de las cortinas, lo deberá agujerear y, posteriormente, introducirlo y romperlo en el lugar sagrado. (6) Un recipiente de cobre que salió de las cortinas, deberá ser introducido de nuevo, limpiado y enjuagado en lugar sagrado. Si se impurificó fuera de las cortinas, le hará un gran orificio y, después, lo introducirá y lo lavará en un lugar sagrado. (7) Tanto si se cocinó en el recipiente, como si se volcó en él el cocido hirviendo, tanto si se trata de carne de Kodshé-Kadashim, como de carne de Kadashim-Kalim, se requiere en todos ellos lavado y enjuagado. Rabí Shimón dice: Los Kadashim-Kalim no requieren lavado y enjuagado. Rabí Tarfón dice: Si se cocinó en él desde el comienzo de la festividad, puede seguir haciéndolo durante toda la festividad. Los sabios alegan: Todo el tiempo que esté permitido comer dicho sacrificio. Limpieza y enjuague: Lavado como la de un vaso y enjuague como el de un vaso. El lavado con agua caliente y el enjuague con agua fría. El asador y la parrilla los introducirá en agua caliente. (8) Si cocinó en él carne sagrada y carne profana, o bien, Kodshé-Kadashim y Kadashim-Kalim, si hay lo suficiente como para dar sabor, los de menos santidad seguirán las normas para su ingestión de los más sagrados, y no necesitan los recipientes ser lavados ni enjuagados, y no invalidan con el contacto. Si una torta toca a otra, o un trozo de carne a otro, no invalidan a todas las tortas o a todos los trozos de carne, sino que se prohíbe sólo el lugar del contacto.

    Rabí Ovadiá de Bartenura

    (1) LA SANGRE DE UN SACRIFICIO JATAT. Tal como versa: “Quien salpique de su sangre...será lavada en lugar sagrado” (Levítico 6:20). “EN UN LUGAR SAGRADO SERÁ INGERIDO”. Con respecto a ese asunto está escrito. LA QUE SE QUEDA ADENTRO. Su sangre requiere ser rociada en el interior y no son ingeridas. “LA LEY DEL JATAT”. Al principio de este tema está mencionado. UNA SOLA LEY PARA TODOS LOS JATAT. Se refiere, precisamente, a un Jatat practicado en un animal, pero la sangre de un ave Jatat no requiere ser lavada, por cuanto está escrito: “será degollado el Jatat”, a diferencia del ave que no es degollada. (2) NO REQUIERE SER LAVADA. Debido a que está escrito: “Quien salpique de su sangre”, de su sangre Kasher, no de sangre no apta. ERA KASHER. En el momento de la rociadura. CUANDO PASÓ LA NOCHE. La sangre. PERSONAS NO APTAS. Así es el texto correcto, y no “y fue rociada”, ya que incluso un Jatat Kasher cuya sangre salpicó sobre una prenda después de la rociadura, no requiere ser lavada, tal como se dijo: “quien salpique de su sangre”, con la excepción de aquella que ya fue rociada. (3) SALIÓ SALPICADA DESDE SU CUELLO. Del animal. NO SE EXIGE QUE SEA LAVADA. Ya que está escrito: “quien salpique”, o sea, sólo aquella que se puede rociar. DESDE UN VÉRTICE. Del Altar. DESDE LA BASE. De los restos de sangre que iban a ser vertidos sobre la base pero aún no lo fueron. Siendo que ya se roció la sangre debida, el resto no requiere lavado, tal como versa: “quien salpique”, con la excepción de aquella que ya hubo rociamiento de su sangre. RECOGIDA EN EL RECIPIENTE. Aquí explica cuál es la razón, o sea, por qué una sangre que cayó al suelo y fue recogida, si manchó, no requiere ser lavada. Y contesta: porque sólo requiere lavado aquella que fue depositada primeramente en el recipiente. DESTINADA A SER SALPICADA. Esto excluye haber introducido una cantidad insuficiente para el rociamiento en un recipiente y otra cantidad igual en otro, y después juntarlas, ya que este procedimiento excluye del lavado. ANTES DE SER DESPELLEJADO, NO REQUIERE LAVADO. Pues está escrito: “sobre la prenda”, sabiendo que una prenda, por muy pequeña que sea, si tiene nombre de “prenda”, es que posee la cualidad de poder impurificarse. Igualmente, todo lo que puede contagiarse de la impureza, y, sabemos, que mientras no fue despellejado no puede contagiarse. Sin embargo, después de despellejada sí puede contagiarse, siempre y cuando pretendiera con ella utilizarla como cobertor para una carroza o como cubierta para una cama. Esta no requiera ser pelada. DESPUÉS DE DESPELLEJADO. Mientras que no sea destinada la piel para alguna función. NO REQUIERE SER LAVADA. Se requiere algo que pueda contagiarse de impurezas, aún sin intención. EL LUGAR EXACTO DE LA PRENDA. Y no toda la prenda. PUEDE CONTAGIARSE DE LA IMPUREZA. Aún sin intención. No obstante debe haber habido una acción, tal como dijo Rabí Yehudá. Así se fijó la Ley. AQUELLO QUE SE PUEDE LAVAR. Esto excluye un recipiente de madera, que aunque sí puede contagiarse de impurezas no se puede lavar, la sangre resbala en él, no necesita ser lavado. (4) EN EL LUGAR SAGRADO. En el Atrio (Azará). LA ROTURA DEL RECIPIENTE DE ARCILLA. Después que escribió “será lavado en un lugar sagrado”, sigue diciendo: “Un recipiente de arcilla usado para cocinar, será quebrado”, o sea, comparó la rotura de un recipiente de arcilla a su lavado. Al igual que el lavado se hace en un lugar sagrado, también la rotura debe ser allí. ESTE TEMA. Se refiere al lavado. (5) LA PRENDA. Sobre la cual salpicó sangre de un Jatat, y salió de las cortinas. SI SE IMPURIFICÓ FUERA DE LAS CORTINAS. Después de haber salido, ya no se permite que introduzca la impureza a la Azará. LA RASGARÁ. En su mayor parte. Así queda purificada de su impureza. LA INTRODUCIRÁ. A pesar de que una prenda que se impurificó y fue rasgada en su mayor parte, aún conserva su impureza por determinación rabínica, hasta que el lugar de unión no supere el ancho de un pañuelo, aquí permitieron que sea introducido a la Azará para que se cumpla el precepto de “lavado”. Esto, a pesar de que, según la Torá, ya se purificó después de rasgarse su mayor parte. LO DEBERÁ AGUJEREAR. Para purificarlo de su impureza. Debe ser un agujero, como mínimo del tamaño de una pequeña raíz, así logra purificarlo y mantenerlo servible como recipiente. De esta forma cumple con el precepto de romperlo dentro del Templo. Sin embargo, si le hizo un orificio grande, deja de considerarse recipiente y ya no puede introducirlo para romperlo. La Torá dijo: “Un recipiente de arcilla que sirvió para guisar debe ser quebrado”, o sea, a la hora de romperlo debe ser considerado aún como recipiente. Sin embargo, este, cuando fue llevado a romper ya no era considerado recipiente. (6) LE HARÁ UN GRAN ORIFICIO. Un recipiente de metal no se purifica de su impureza con un pequeño orificio, sino con uno grande. No obstante, si alguien le hizo un gran orificio, deberá golpearlo con un martillo hasta cerrar el agujero, así adquirirá de nuevo el nombre de “recipiente”, ya que, a la hora del lavado, deberá ser de hecho un recipiente. (7) SI SE VOLCÓ EN ÉL. Está escrito: “Un recipiente de arcilla, en el cual se cocinó, será quebrado”. Se dijeron las palabras “en el cual” seguido de “cocinó”, y no “Si en un recipiente de arcilla se cocinara, deberá quebrarse”, para hacer constar que solo con absorber algo las paredes de este, se debe romper. NO REQUIEREN LAVADO Y ENJUAGADO. Aún así, Rabí Shimón opina que deberán ser sumergidos en agua hirviendo, ya que aquello que absorbieron las paredes se considera Notar (sobrante), y va a ser expulsado después de un tiempo si no lo hierve. Sólo el lavado y el enjuague es lo que excluye Rabí Shimón en los Kadashim-Kalim. El lavado y el enjuague de los Kodshé-Kadashim, debe hacerse sólo con agua, no con vino o con la mezcla de agua-vino. Si utilizó sólo una parte del recipiente para cocinar, deberá ser lavado y enjuagado enteramente, el lavado en caliente y el enjuague en frío. Sin embargo, los de Kadashim-Kalim, según Rabí Shimón, pueden ser hervidos en vino o incluso con agua-vino hirviendo. Además, no se exige introducirlo todo en agua hirviendo, sino sólo la parte afectada. Después del hervido no es necesario enjuagarlo en frío, porque todo aquello que fue mencionado por el texto sagrado, se refiere al Jatat y a los Kodshé-Kodashim, y no a los Kodashim-Kalim. La Halajá no se determinó como dijo Rabí Shimón. La razón es que, solamente, la Terumá se excluye de todo lo antedicho, tal como versa: “Ella será ingerida” (Levítico 6:22). Sobre esto hay una Baraita que dice: Está escrito “ella”, para excluir a la Terumá que no debe pasar por el lavado y el enjuague como los Kadashim. Por otro lado, los Kadashim-Kalim igualan en todo a los Kodshé-Kadashim en lo referido al lavado y enjuague. Sobre el significado exacto de los conceptos “lavado” y “enjuague” divergen los Tanaítas de la Baraita. Hay quien dice que “lavado” es introducción en agua hirviendo y “enjuague”, en frío. Otro dice que el lavado y el enjuague deben hacerse con agua fría, después de la Hagalá, tal como se menciona seguidamente en la Mishná, sólo que, el lavado se hace como a un vaso y el enjuagado igual, o sea, uno por dentro y otro por fuera. Eso es, el enjuague por fuera y el lavado por dentro. La Halajá se practica como aquel que dijo que el lavado es en caliente y el enjuague en frío. DURANTE TODA LA FESTIVIDAD. Sin necesidad de lavar y enjuagar, pero al final lo lavará y enjuagará ya que, lo de un día se ensucia al siguiente. Esto ocurre porque, siendo que los Shelamim son muy numerosos durante la festividad, lo que absorben las paredes no se denomina Notar. El tiempo para la ingestión de los Shelamim es de dos días, por tanto, si hoy cocina un Shelamim y mañana otro de los degollados al día siguiente, lo que se absorbió hoy lo expulsa mañana y así sucesivamente. De esta forma no se da el caso de Notar. TODO EL TIEMPO QUE ESTÉ PERMITIDO COMER. O sea, que no haya entre el final del cocido y el comienzo del lavado y enjuague, solamente el tiempo que dura la ingestión, nada más. Así está escrito: “será lavado y enjuagado”, y seguidamente, “todo varón entre los Cohanim lo comerá”. La razón por la cual fueron escritos estos dos textos juntos, es para decirnos que sólo puede esperar el tiempo dado para su ingestión. Después deberá lavarlo y enjuagarlo enseguida, en el mismo día. La Halajá se determinó según la opinión de los Sabios. LA PARRILLA. “Gradila” en idioma foráneo. Es una especie de reja donde se asa la carne. LAS INTRODUCIRÁ. En agua caliente. (8) SEGUIRÁN LAS NORMAS PARA SU INGESTIÓN DE LOS MÁS SAGRADOS. La Mishná contiene un error en el texto. Así es correcto: “si hay lo suficiente como para dar sabor, los de menos santidad seguirán las normas para su ingestión de los más sagrados, en el lado interior de las cortinas, durante un día y una noche, y deberá ser lavados y enjuagados e invalidan con el contacto. Si no hay lo suficiente como para dar sabor, entonces, los de menos santidad no siguen las normas de los que más tienen, además de no requerir lavado y enjuague, y no invalidar con el contacto. Esta Mishná es atribuida a Rabí Shimón, quien ya dijo que los Kadashim-Kalim no requieren ni lavado ni enjuague. TORTA. Se refiere a una torta Kasher que contactó con una torta Pasul y se le pega algo de ella.

    Tratado Zebajim 115b

    Estudiamos en una Baraita: “Y también los Cohanim que se acerquen al Eterno serán consagrados” (Éxodo 19:22). Rabí Yehoshúa Ben-Korjá dice: Esto era una advertencia para los primogénitos que no debían subir al monte Sinai. Ribí dice: Esta advertencia iba dirigida a Nadab y Abihú. Esta idea puede concordar con aquel que dijo que era una advertencia hacia Nadab y Abihú, ya que está escrito: “Eso fue lo que el Eterno expresó: Seré consagrado entre los más cercanos” (Levítico 10:3). Sin embargo, aquel que dijo que la advertencia era referida a los primogénitos, ¿dónde existe alguna referencia que lo demuestre? Está escrito: “Me presentaré allí ante los Hijos de Israel, y se consagrará por Mi Honor” (Éxodo 29:43). No leas “por Mi Honor”, sino, “por los que Me honran”. Este concepto se lo insinuó el Eterno a Moshé, pero no supo de qué se trataba hasta que murieron los hijos de Aharón. Cuando perecieron los hijos de Aharón, le dijo: Hermano, tus hijos murieron para santificar el Nombre del Eterno. Cuando supo Aharón que sus hijos fueron los elegidos por el Eterno, calló recibiendo su recompensa, tal como versa: “Y calló Aharón” (Levítico 10:3). Igualmente, con referencia a David se dijo: “Calla ante el Eterno, y confía en Él” (Salmos 37:7). Queriendo decir: Aunque haga que caigan de los tuyos muchos cadáveres, permanece callado. Con respecto a Shlomó también se dijo: “Tiempo para callar y tiempo para hablar” (Eclesiastés 3:7). A veces, el hombre, callando, recibe recompensa por su silencio, y a veces, recibe su recompensa por hablar. Esto fue lo que dijo Rabí Jiyá Bar Abá en nombre de Rabí Yojanán: ¿Qué significa el párrafo: “Temible eres, Todopoderoso, desde Tus santuarios” (Salmos 68:36). No leas “desde Tus santuarios”, sino, “con los que Te santifican”. Cuando el Santo, Bendito Él, juzga a Sus santos, es temido, elevado y alabado.

Unos representan la misericordia y otros, el rigor del Juicio. Asimismo, todo lo que hizo en Egipto, lo realizó de la misma forma. Por eso está escrito: “te he comparado, amada mía”, igual que ella, para asesinar. Ven y observa lo que está escrito: “Y se asentó Israel en la Tierra de Egipto, en la Tierra de Goshen, y se posesionaron de ella, y fructificaron y se multiplicaron en gran medida” (Génesis 47:27). Una posesión para la eternidad. “Se posesionaron de ella”, porque son dignos de ello, como ya aclaramos. “Y fructificaron y se multiplicaron...”, de cierto, puesto que no fueron sometidos a ningún tormento.     Rambam, Leyes sobre el Arrepentimiento, Capítulo 2

    (1) Quien se confiesa verbalmente pero no resolvió en su corazón abandonar el pecado, se asemeja a aquel que se sumerge en un Mikvé mientras agarra un reptil en su mano, donde la inmersión no le sirve hasta que no suelte dicho reptil. Asimismo está dicho: “Quien confiesa y abandona, obtendrá misericordia” (Proverbios 28:13). Se debe especificar el pecado, por cuanto está dicho: “Este pueblo ha cometido un gran pecado, y se han hecho dioses de oro” (Éxodo 32:31). (2) Uno de los caminos hacia el arrepentimiento consiste en que el arrepentido debe clamar continuamente ante el Eterno con llanto y súplicas, practicando la caridad según sus posibilidades, y alejándose lo más posible de aquello en qué pecó, además, debe cambiar su nombre, queriendo significar: “Soy otro, no soy el mismo hombre que cometió aquellos pecados”. Cambiará todos su actos hacia el bien y el camino recto. También, se exiliará a otro lugar, puesto que el exilio expía los pecados, ya que le ocasiona humillación y lo lleva a ser sumiso y humilde. (3) Es sumamente meritorio que el pecador se confiese públicamente y de a conocer sus pecados, y que revele las transgresiones cometidas contra otras personas, diciéndoles: “He pecado contra Fulano y le hice esto y aquello. Sin embargo, hoy me arrepiento”. Quien, por ser orgulloso no lo publica y oculta sus pecados, no se ha arrepentido totalmente, tal como versa: “Quien encubre sus pecados no tendrá éxito” (Proverbios 28:13).

    Moral

    Séfer Hajaredim, 71a

    El corazón que medita sobre sus asuntos personales en la hora de la oración o la repetición que hace el oficiante, porta un grave pecado. Sobre él se dijo: “Y los que Me desprecian serán deshonrados” (Samuel I 2:30). desprecie más Su Nombre que él, porque delante de un rey, e incluso, delante de uno de sus ministros, con temor y reverencia todo hombre se comporta, temiendo no ser atrapado en sus propias palabras, pues, “La ira de un rey es como mensajero de muerte” (Proverbios 16:14). Cuánto y más la Ira del Rey de reyes, el Santo, Bendito Él, del Cual, incluso Moshé Rabenu temió, tal como versa: “Temo debido a la ira y a la furia” (Deuteronomio 9:19). Además, su pecado es grave por haber testimoniado falsamente, ya que proclamó en la Keriat Shemá: “Y Lo serviréis con todos vuestros corazones” (Deuteronomio 11:13). Se refiere al servicio del corazón, o sea, la plegaria (Tratado Taanit 2a), y, he aquí, que su corazón se desvía del eterno hacia pensamientos vanos. Sobre él dirá el eterno: “A Mí me abandonaron, Fuente de Agua Viva, para cavar pozos, pozos agrietados” (Jeremías 2:13). Y, adicionalmente, está escrito: “Cerca estás de sus bocas, pero lejos de sus entrañas” (Ibíd. 12:2). Y está escrito: “Con su boca y sus labios Me honró, mas su corazón se alejó de Mí” (Isaías 29:13). Y está escrito: “Si soy el Padre, ¿dónde está Mi Honor? Si soy el Amo, ¿dónde está el Temor hacia Mí? (Malaquías 1:6). Leemos en la oración: “¡Padre Misericordioso, escucha nuestra voz!”, y en el comienzo de ella: “Eterno, abre mis labios”, para servirle con amor, como a un Padre, y, con temor, como a un Amo.
